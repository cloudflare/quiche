27/// TLS credentials to authenticate the endpoint.
28#[derive(Clone, Copy, Debug)]
29pub struct TlsCertificatePaths<'p> {
30    /// Path to the endpoint's TLS certificate.
31    pub cert: &'p str,
32    /// Path to the endpoint's private key.
33    pub private_key: &'p str,
34    /// `cert`'s PKI certificate type.
35    pub kind: CertificateKind,
36}
37
38/// Types of PKI certificates supported by the crate.
39#[derive(Default, Clone, Copy, Debug, PartialEq, Eq, PartialOrd, Ord, Hash)]
40pub enum CertificateKind {
41    /// Standard X509 TLS certificate.
42    #[default]
43    X509,
44    /// [Raw public key] TLS certificate.
45    ///
46    ///
47    /// [Raw public key]: https://datatracker.ietf.org/doc/html/rfc7250
48    RawPublicKey,
49}
