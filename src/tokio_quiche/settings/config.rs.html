38/// Whether `--cfg capture_keylogs` was set at build time. We keep supporting
39/// the `capture_keylogs` feature for backward compatibility.
40const KEYLOGFILE_ENABLED: bool =
41    cfg!(capture_keylogs) || cfg!(feature = "capture_keylogs");
42
43/// Internal representation of the combined configuration for a QUIC connection.
44pub(crate) struct Config {
45    pub quiche_config: quiche::Config,
46    pub disable_client_ip_validation: bool,
47    pub qlog_dir: Option<String>,
48    pub has_gso: bool,
49    pub pacing_offload: bool,
50    pub enable_expensive_packet_count_metrics: bool,
51    pub keylog_file: Option<File>,
52    pub listen_backlog: usize,
53    pub handshake_timeout: Option<Duration>,
54    pub has_ippktinfo: bool,
55    pub has_ipv6pktinfo: bool,
56}
57
58impl AsMut<quiche::Config> for Config {
59    fn as_mut(&mut self) -> &mut quiche::Config {
60        &mut self.quiche_config
61    }
62}
63
64impl Config {
65    pub(crate) fn new(
66        params: &ConnectionParams, socket_capabilities: SocketCapabilities,
67    ) -> QuicResult<Self> {
68        let quic_settings = &params.settings;
69        let keylog_path = match &quic_settings.keylog_file {
70            Some(f) => Some(Cow::Borrowed(f.as_ref())),
71            None => std::env::var_os("SSLKEYLOGFILE").map(Cow::from),
72        };
73        let keylog_file = keylog_path.and_then(|path| if KEYLOGFILE_ENABLED {
74                File::options().create(true).append(true).open(path)
75                    .inspect_err(|e| log::warn!("failed to open SSLKEYLOGFILE"; "error" => e))
76                    .ok()
77            } else {
78                log::warn!("SSLKEYLOGFILE is set, but `--cfg capture_keylogs` was not enabled. No keys will be logged.");
79                None
80            });
81
82        let SocketCapabilities {
83            has_gso,
84            has_txtime: pacing_offload,
85            has_ippktinfo,
86            has_ipv6pktinfo,
87            ..
88        } = socket_capabilities;
89
90        #[cfg(feature = "gcongestion")]
91        let pacing_offload = quic_settings.enable_pacing && pacing_offload;
92
93        Ok(Config {
94            quiche_config: make_quiche_config(params, keylog_file.is_some())?,
95            disable_client_ip_validation: quic_settings
96                .disable_client_ip_validation,
97            qlog_dir: quic_settings.qlog_dir.clone(),
98            has_gso,
99            pacing_offload,
100            enable_expensive_packet_count_metrics: quic_settings
101                .enable_expensive_packet_count_metrics,
102            keylog_file,
103            listen_backlog: quic_settings.listen_backlog,
104            handshake_timeout: quic_settings.handshake_timeout,
105            has_ippktinfo,
106            has_ipv6pktinfo,
107        })
108    }
109}

111fn make_quiche_config(
112    params: &ConnectionParams, should_log_keys: bool,
113) -> QuicResult<quiche::Config> {
114    let ssl_ctx_builder = params
115        .hooks
116        .connection_hook
117        .as_ref()
118        .zip(params.tls_cert)
119        .and_then(|(hook, tls)| hook.create_custom_ssl_context_builder(tls));
120
121    let mut config = if let Some(builder) = ssl_ctx_builder {
122        quiche::Config::with_boring_ssl_ctx_builder(
123            quiche::PROTOCOL_VERSION,
124            builder,
125        )?
126    } else {
127        quiche_config_with_tls(params.tls_cert)?
128    };
129
130    let quic_settings = &params.settings;
131
132    let alpns: Vec<&[u8]> =
133        quic_settings.alpn.iter().map(Vec::as_slice).collect();
134    config.set_application_protos(&alpns).unwrap();
135
136    if let Some(timeout) = quic_settings.max_idle_timeout {
137        let ms = timeout
138            .as_millis()
139            .try_into()
140            .map_err(|_| "QuicSettings::max_idle_timeout exceeds u64")?;
141        config.set_max_idle_timeout(ms);
142    }
143
144    config.enable_dgram(
145        quic_settings.enable_dgram,
146        quic_settings.dgram_recv_max_queue_len,
147        quic_settings.dgram_send_max_queue_len,
148    );
149
150    config.set_max_recv_udp_payload_size(quic_settings.max_recv_udp_payload_size);
151    config.set_max_send_udp_payload_size(quic_settings.max_send_udp_payload_size);
152    config.set_initial_max_data(quic_settings.initial_max_data);
153    config.set_initial_max_stream_data_bidi_local(
154        quic_settings.initial_max_stream_data_bidi_local,
155    );
156    config.set_initial_max_stream_data_bidi_remote(
157        quic_settings.initial_max_stream_data_bidi_remote,
158    );
159    config.set_initial_max_stream_data_uni(
160        quic_settings.initial_max_stream_data_uni,
161    );
162    config.set_initial_max_streams_bidi(quic_settings.initial_max_streams_bidi);
163    config.set_initial_max_streams_uni(quic_settings.initial_max_streams_uni);
164    config.set_disable_active_migration(quic_settings.disable_active_migration);
165    config
166        .set_active_connection_id_limit(quic_settings.active_connection_id_limit);
167    config.set_cc_algorithm_name(quic_settings.cc_algorithm.as_str())?;
168    config.set_initial_congestion_window_packets(
169        quic_settings.initial_congestion_window_packets,
170    );
171    config.set_enable_relaxed_loss_threshold(
172        quic_settings.enable_relaxed_loss_threshold,
173    );
174    config.discover_pmtu(quic_settings.discover_path_mtu);
175    config.set_pmtud_max_probes(quic_settings.pmtud_max_probes);
176    config.enable_hystart(quic_settings.enable_hystart);
177
178    config.enable_pacing(quic_settings.enable_pacing);
179    if let Some(max_pacing_rate) = quic_settings.max_pacing_rate {
180        config.set_max_pacing_rate(max_pacing_rate);
181    }
182
183    if quic_settings.verify_peer {
184        config.verify_peer(quic_settings.verify_peer);
185    }
186
187    config.set_max_connection_window(quic_settings.max_connection_window);
188    config.set_max_stream_window(quic_settings.max_stream_window);
189    config.grease(quic_settings.grease);
190    config.set_max_amplification_factor(quic_settings.max_amplification_factor);
191    config.set_send_capacity_factor(quic_settings.send_capacity_factor);
192    config.set_ack_delay_exponent(quic_settings.ack_delay_exponent);
193    config.set_max_ack_delay(quic_settings.max_ack_delay);
194    config.set_path_challenge_recv_max_queue_len(
195        quic_settings.max_path_challenge_recv_queue_len,
196    );
197    config.set_stateless_reset_token(quic_settings.stateless_reset_token);
198    config.set_disable_dcid_reuse(quic_settings.disable_dcid_reuse);
199
200    if let Some(track_unknown_transport_params) =
201        quic_settings.track_unknown_transport_parameters
202    {
203        config.enable_track_unknown_transport_parameters(
204            track_unknown_transport_params,
205        );
206    }
207    if params.settings.enable_early_data {
208        config.enable_early_data();
209    }
210
211    if should_log_keys {
212        config.log_keys();
213    }
214
215    Ok(config)
216}

218fn quiche_config_with_tls(
219    tls_cert: Option<TlsCertificatePaths>,
220) -> QuicResult<quiche::Config> {
221    let Some(tls) = tls_cert else {
222        return Ok(quiche::Config::new(quiche::PROTOCOL_VERSION).unwrap());
223    };
224
225    match tls.kind {
226        #[cfg(not(feature = "rpk"))]
227        CertificateKind::RawPublicKey => {
228            // TODO: don't compile this enum variant unless rpk feature is enabled
229            panic!("Can't use RPK when compiled without rpk feature");
230        },
231        #[cfg(feature = "rpk")]
232        CertificateKind::RawPublicKey => {
233            let mut ssl_ctx_builder = boring::ssl::SslContextBuilder::new_rpk()?;
234            let raw_public_key = read_file(tls.cert)?;
235            ssl_ctx_builder.set_rpk_certificate(&raw_public_key)?;
236
237            let raw_private_key = read_file(tls.private_key)?;
238            let pkey =
239                boring::pkey::PKey::private_key_from_pem(&raw_private_key)?;
240            ssl_ctx_builder.set_null_chain_private_key(&pkey)?;
241
242            Ok(quiche::Config::with_boring_ssl_ctx_builder(
243                quiche::PROTOCOL_VERSION,
244                ssl_ctx_builder,
245            )?)
246        },
247        CertificateKind::X509 => {
248            let mut config =
249                quiche::Config::new(quiche::PROTOCOL_VERSION).unwrap();
250            config.load_cert_chain_from_pem_file(tls.cert)?;
251            config.load_priv_key_from_pem_file(tls.private_key)?;
252            Ok(config)
253        },
254    }
255}

257#[cfg(feature = "rpk")]
258fn read_file(path: &str) -> QuicResult<Vec<u8>> {
259    use anyhow::Context as _;
260    std::fs::read(path)
261        .with_context(|| format!("read {path}"))
262        .map_err(Into::into)
