37const U64_SZ: usize = std::mem::size_of::<u64>();
38const MAX_CONN_ID_QUADS: usize = MAX_CONN_ID_LEN.div_ceil(U64_SZ);
39const CONN_ID_USABLE_LEN: usize = min_usize(
40    // Last byte in CidOwned::Optimized stores CID length
41    MAX_CONN_ID_QUADS * U64_SZ - 1,
42    // CID length must fit in 1 byte
43    min_usize(MAX_CONN_ID_LEN, u8::MAX as _),
44);
45
46const fn min_usize(v1: usize, v2: usize) -> usize {
47    if v1 < v2 {
48        v1
49    } else {
50        v2
51    }
52}

54/// A non unique connection identifier, multiple Cids can map to the same
55/// conenction.
56#[derive(PartialEq, Eq, PartialOrd, Ord)]
57enum CidOwned {
58    /// The QUIC connections IDs theoretically have unbounded length, so for the
59    /// generic case a boxed slice is used to store the ID.
60    Generic(Box<[u8]>),
61    /// For QUIC version 1 (the one that actually exists) the maximal ID size is
62    /// `20`, which should correspond to the `MAX_CONN_ID_LEN` value. For
63    /// this common case, we store the ID in a u64 array for faster
64    /// comparison (and therefore BTreeMap lookups).
65    Optimized([u64; MAX_CONN_ID_QUADS]),
66}
67
68impl From<&ConnectionId<'_>> for CidOwned {
69    #[inline(always)]
70    fn from(value: &ConnectionId<'_>) -> Self {
71        if value.len() > CONN_ID_USABLE_LEN {
72            return CidOwned::Generic(value.as_ref().into());
73        }
74
75        let mut cid = [0; MAX_CONN_ID_QUADS];
76
77        value
78            .chunks(U64_SZ)
79            .map(|c| match c.try_into() {
80                Ok(v) => u64::from_le_bytes(v),
81                Err(_) => {
82                    let mut remainder = [0u8; U64_SZ];
83                    remainder[..c.len()].copy_from_slice(c);
84                    u64::from_le_bytes(remainder)
85                },
86            })
87            .enumerate()
88            .for_each(|(i, v)| cid[i] = v);
89
90        // In order to differentiate cids with zeroes as opposed to shorter cids,
91        // append the cid length.
92        *cid.last_mut().unwrap() |= (value.len() as u64) << 56;
93
94        CidOwned::Optimized(cid)
95    }
96}

98/// A map for QUIC connections.
99///
100/// Due to the fact that QUIC connections can be identified by multiple QUIC
101/// connection IDs, we have to be able to map multiple IDs to the same
102/// connection.
103#[derive(Default)]
104pub(crate) struct ConnectionMap {
105    quic_id_map: BTreeMap<CidOwned, mpsc::Sender<Incoming>>,
106}
107
108impl ConnectionMap {
109    pub(crate) fn insert<Tx, M>(
110        &mut self, cid: &ConnectionId<'_>, conn: &InitialQuicConnection<Tx, M>,
111    ) where
112        Tx: DatagramSocketSend + Send + 'static,
113        M: Metrics,
114    {
115        let ev_sender = conn.incoming_ev_sender.clone();
116        self.quic_id_map.insert(cid.into(), ev_sender);
117    }
118
119    pub(crate) fn map_cid(
120        &mut self, existing_cid: &ConnectionId<'_>, new_cid: &ConnectionId<'_>,
121    ) {
122        if let Some(ev_sender) = self.quic_id_map.get(&existing_cid.into()) {
123            self.quic_id_map.insert(new_cid.into(), ev_sender.clone());
124        }
125    }
126
127    pub(crate) fn unmap_cid(&mut self, cid: &ConnectionId<'_>) {
128        self.quic_id_map.remove(&cid.into());
129    }
130
131    pub(crate) fn get(
132        &self, id: &ConnectionId,
133    ) -> Option<&mpsc::Sender<Incoming>> {
134        if id.len() == MAX_CONN_ID_LEN {
135            // Although both branches run the same code, the one here will
136            // generate an optimized version for the length we are
137            // using, as opposed to temporary cids sent by clients.
138            self.quic_id_map.get(&id.into())
139        } else {
140            self.quic_id_map.get(&id.into())
141        }
142    }
143}

145#[cfg(test)]
146mod tests {
147    use super::*;
148    use quiche::ConnectionId;
149
150    #[test]
151    fn cid_storage() {
152        let max_v1_cid = ConnectionId::from_ref(&[0xfa; MAX_CONN_ID_LEN]);
153        let optimized = CidOwned::from(&max_v1_cid);
154        assert!(
155            matches!(optimized, CidOwned::Optimized(_)),
156            "QUIC v1 CID is not stored inline"
157        );
158
159        let oversize_cid = ConnectionId::from_ref(&[0x1b; MAX_CONN_ID_LEN + 20]);
160        let boxed = CidOwned::from(&oversize_cid);
161        assert!(
162            matches!(boxed, CidOwned::Generic(_)),
163            "Oversized CID is not boxed"
164        );
165    }
166}
