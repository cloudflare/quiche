211    #[cfg(windows)]
212    fn load_ca_certs(&mut self) -> Result<()> {
213        unsafe {
214            let cstr = ffi::CString::new("Root").map_err(|_| Error::TlsFail)?;
215            let sys_store =
216                windows_sys::Win32::Security::Cryptography::CertOpenSystemStoreA(
217                    0,
218                    cstr.as_ptr() as windows_sys::core::PCSTR,
219                );
220            if sys_store.is_null() {
221                return Err(Error::TlsFail);
222            }
223
224            let ctx_store = SSL_CTX_get_cert_store(self.as_mut_ptr());
225            if ctx_store.is_null() {
226                return Err(Error::TlsFail);
227            }
228
229            let mut ctx_p = windows_sys::Win32::Security::Cryptography::CertEnumCertificatesInStore(
230                sys_store,
231                ptr::null(),
232            );
233
234            while !ctx_p.is_null() {
235                let in_p = (*ctx_p).pbCertEncoded as *const u8;
236
237                let cert = d2i_X509(
238                    ptr::null_mut(),
239                    &in_p,
240                    (*ctx_p).cbCertEncoded as i32,
241                );
242                if !cert.is_null() {
243                    X509_STORE_add_cert(ctx_store, cert);
244                }
245
246                X509_free(cert);
247
248                ctx_p = windows_sys::Win32::Security::Cryptography::CertEnumCertificatesInStore(
249                    sys_store, ctx_p,
250                );
251            }
252
253            // tidy up
254            windows_sys::Win32::Security::Cryptography::CertFreeCertificateContext(ctx_p);
255            windows_sys::Win32::Security::Cryptography::CertCloseStore(
256                sys_store, 0,
257            );
258        }
259
260        Ok(())
261    }

294    pub fn set_alpn(&mut self, v: &[&[u8]]) -> Result<()> {
295        let mut protos: Vec<u8> = Vec::new();
296
297        for proto in v {
298            protos.push(proto.len() as u8);
299            protos.extend_from_slice(proto);
300        }
301
302        // Configure ALPN for servers.
303        unsafe {
304            SSL_CTX_set_alpn_select_cb(
305                self.as_mut_ptr(),
306                Some(select_alpn),
307                ptr::null_mut(),
308            );
309        }
310
311        // Configure ALPN for clients.
312        map_result_zero_is_success(unsafe {
313            SSL_CTX_set_alpn_protos(
314                self.as_mut_ptr(),
315                protos.as_ptr(),
316                protos.len(),
317            )
318        })
319    }

535    pub fn do_handshake(&mut self, ex_data: &mut ExData) -> Result<()> {
536        self.set_ex_data(*QUICHE_EX_DATA_INDEX, ex_data)?;
537        let rc = unsafe { SSL_do_handshake(self.as_mut_ptr()) };
538        self.set_ex_data::<Connection>(*QUICHE_EX_DATA_INDEX, ptr::null())?;
539
540        self.set_transport_error(ex_data, rc);
541        self.map_result_ssl(rc)
542    }
543
544    pub fn process_post_handshake(&mut self, ex_data: &mut ExData) -> Result<()> {
545        // If SSL_provide_quic_data hasn't been called since we last called
546        // SSL_process_quic_post_handshake, then there's nothing to do.
547        if !self.provided_data_outstanding {
548            return Ok(());
549        }
550        self.provided_data_outstanding = false;
551
552        self.set_ex_data(*QUICHE_EX_DATA_INDEX, ex_data)?;
553        let rc = unsafe { SSL_process_quic_post_handshake(self.as_mut_ptr()) };
554        self.set_ex_data::<Connection>(*QUICHE_EX_DATA_INDEX, ptr::null())?;
555
556        self.set_transport_error(ex_data, rc);
557        self.map_result_ssl(rc)
558    }

599    fn map_result_ssl(&mut self, bssl_result: c_int) -> Result<()> {
600        match bssl_result {
601            1 => Ok(()),
602
603            _ => {
604                let ssl_err = self.get_error(bssl_result);
605                match ssl_err {
606                    // SSL_ERROR_SSL
607                    1 => {
608                        log_ssl_error();
609
610                        Err(Error::TlsFail)
611                    },
612
613                    // SSL_ERROR_WANT_READ
614                    2 => Err(Error::Done),
615
616                    // SSL_ERROR_WANT_WRITE
617                    3 => Err(Error::Done),
618
619                    // SSL_ERROR_WANT_X509_LOOKUP
620                    4 => Err(Error::Done),
621
622                    // SSL_ERROR_SYSCALL
623                    5 => Err(Error::TlsFail),
624
625                    // SSL_ERROR_PENDING_SESSION
626                    11 => Err(Error::Done),
627
628                    // SSL_ERROR_PENDING_CERTIFICATE
629                    12 => Err(Error::Done),
630
631                    // SSL_ERROR_WANT_PRIVATE_KEY_OPERATION
632                    13 => Err(Error::Done),
633
634                    // SSL_ERROR_PENDING_TICKET
635                    14 => Err(Error::Done),
636
637                    // SSL_ERROR_EARLY_DATA_REJECTED
638                    15 => {
639                        self.reset_early_data_reject();
640                        Err(Error::Done)
641                    },
642
643                    // SSL_ERROR_WANT_CERTIFICATE_VERIFY
644                    16 => Err(Error::Done),
645
646                    _ => Err(Error::TlsFail),
647                }
648            },
649        }
650    }
651
652    fn set_transport_error(&mut self, ex_data: &mut ExData, bssl_result: c_int) {
653        // SSL_ERROR_SSL
654        if self.get_error(bssl_result) == 1 {
655            // SSL_ERROR_SSL can't be recovered so ensure we set a
656            // local_error so the connection is closed.
657            // See https://www.openssl.org/docs/man1.1.1/man3/SSL_get_error.html
658            if ex_data.local_error.is_none() {
659                *ex_data.local_error = Some(ConnectionError {
660                    is_app: false,
661                    error_code: INTERNAL_ERROR,
662                    reason: Vec::new(),
663                })
664            }
665        }
666    }

724        // and the associated data is always `ExData`.
725        let idx: Index<boring::ssl::Ssl, ExData> =
726            unsafe { Index::from_raw(*QUICHE_EX_DATA_INDEX) };
727
728        ssl.ex_data_mut(idx)
729    }

739fn get_cipher_from_ptr(cipher: *const SSL_CIPHER) -> Result<crypto::Algorithm> {
740    let cipher_id = unsafe { SSL_CIPHER_get_id(cipher) };
741
742    let alg = match cipher_id {
743        0x0300_1301 => crypto::Algorithm::AES128_GCM,
744        0x0300_1302 => crypto::Algorithm::AES256_GCM,
745        0x0300_1303 => crypto::Algorithm::ChaCha20_Poly1305,
746        _ => return Err(Error::TlsFail),
747    };
748
749    Ok(alg)
750}

752extern "C" fn set_read_secret(
753    ssl: *mut SSL, level: crypto::Level, cipher: *const SSL_CIPHER,
754    secret: *const u8, secret_len: usize,
755) -> c_int {
756    let ex_data = match ExData::from_ssl_ptr(ssl) {
757        Some(v) => v,
758
759        None => return 0,
760    };
761
762    trace!("{} set read secret lvl={:?}", ex_data.trace_id, level);
763
764    let space = match level {
765        crypto::Level::Initial => &mut ex_data.crypto_ctx[packet::Epoch::Initial],
766        crypto::Level::ZeroRTT =>
767            &mut ex_data.crypto_ctx[packet::Epoch::Application],
768        crypto::Level::Handshake =>
769            &mut ex_data.crypto_ctx[packet::Epoch::Handshake],
770        crypto::Level::OneRTT =>
771            &mut ex_data.crypto_ctx[packet::Epoch::Application],
772    };
773
774    let aead = match get_cipher_from_ptr(cipher) {
775        Ok(v) => v,
776
777        Err(_) => return 0,
778    };
779
780    // 0-RTT read secrets are present only on the server.
781    if level != crypto::Level::ZeroRTT || ex_data.is_server {
782        let secret = unsafe { slice::from_raw_parts(secret, secret_len) };
783
784        let open = match crypto::Open::from_secret(aead, secret) {
785            Ok(v) => v,
786
787            Err(_) => return 0,
788        };
789
790        if level == crypto::Level::ZeroRTT {
791            space.crypto_0rtt_open = Some(open);
792            return 1;
793        }
794
795        space.crypto_open = Some(open);
796    }
797
798    1
799}

801extern "C" fn set_write_secret(
802    ssl: *mut SSL, level: crypto::Level, cipher: *const SSL_CIPHER,
803    secret: *const u8, secret_len: usize,
804) -> c_int {
805    let ex_data = match ExData::from_ssl_ptr(ssl) {
806        Some(v) => v,
807
808        None => return 0,
809    };
810
811    trace!("{} set write secret lvl={:?}", ex_data.trace_id, level);
812
813    let space = match level {
814        crypto::Level::Initial => &mut ex_data.crypto_ctx[packet::Epoch::Initial],
815        crypto::Level::ZeroRTT =>
816            &mut ex_data.crypto_ctx[packet::Epoch::Application],
817        crypto::Level::Handshake =>
818            &mut ex_data.crypto_ctx[packet::Epoch::Handshake],
819        crypto::Level::OneRTT =>
820            &mut ex_data.crypto_ctx[packet::Epoch::Application],
821    };
822
823    let aead = match get_cipher_from_ptr(cipher) {
824        Ok(v) => v,
825
826        Err(_) => return 0,
827    };
828
829    // 0-RTT write secrets are present only on the client.
830    if level != crypto::Level::ZeroRTT || !ex_data.is_server {
831        let secret = unsafe { slice::from_raw_parts(secret, secret_len) };
832
833        let seal = match crypto::Seal::from_secret(aead, secret) {
834            Ok(v) => v,
835
836            Err(_) => return 0,
837        };
838
839        space.crypto_seal = Some(seal);
840    }
841
842    1
843}

845extern "C" fn add_handshake_data(
846    ssl: *mut SSL, level: crypto::Level, data: *const u8, len: usize,
847) -> c_int {
848    let ex_data = match ExData::from_ssl_ptr(ssl) {
849        Some(v) => v,
850
851        None => return 0,
852    };
853
854    trace!(
855        "{} write message lvl={:?} len={}",
856        ex_data.trace_id,
857        level,
858        len
859    );
860
861    let buf = unsafe { slice::from_raw_parts(data, len) };
862
863    let space = match level {
864        crypto::Level::Initial => &mut ex_data.crypto_ctx[packet::Epoch::Initial],
865        crypto::Level::ZeroRTT => unreachable!(),
866        crypto::Level::Handshake =>
867            &mut ex_data.crypto_ctx[packet::Epoch::Handshake],
868        crypto::Level::OneRTT =>
869            &mut ex_data.crypto_ctx[packet::Epoch::Application],
870    };
871
872    if space.crypto_stream.send.write(buf, false).is_err() {
873        return 0;
874    }
875
876    1
877}

912extern "C" fn keylog(ssl: *const SSL, line: *const c_char) {
913    let ex_data = match ExData::from_ssl_ptr(ssl) {
914        Some(v) => v,
915
916        None => return,
917    };
918
919    if let Some(keylog) = &mut ex_data.keylog {
920        let data = unsafe { ffi::CStr::from_ptr(line).to_bytes() };
921
922        let mut full_line = Vec::with_capacity(data.len() + 1);
923        full_line.extend_from_slice(data);
924        full_line.push(b'\n');
925
926        keylog.write_all(&full_line[..]).ok();
927        keylog.flush().ok();
928    }

931extern "C" fn select_alpn(
932    ssl: *mut SSL, out: *mut *const u8, out_len: *mut u8, inp: *mut u8,
933    in_len: c_uint, _arg: *mut c_void,
934) -> c_int {
935    // SSL_TLSEXT_ERR_OK 0
936    // SSL_TLSEXT_ERR_ALERT_WARNING 1
937    // SSL_TLSEXT_ERR_ALERT_FATAL 2
938    // SSL_TLSEXT_ERR_NOACK 3
939
940    // Boringssl internally overwrite the return value from this callback, if the
941    // returned value is SSL_TLSEXT_ERR_NOACK and is quic, then the value gets
942    // overwritten to SSL_TLSEXT_ERR_ALERT_FATAL. In contrast openssl/quictls does
943    // not do that, so we need to explicitly respond with
944    // SSL_TLSEXT_ERR_ALERT_FATAL in case it is needed.
945    // TLS_ERROR is redefined for each vendor.
946    let ex_data = match ExData::from_ssl_ptr(ssl) {
947        Some(v) => v,
948
949        None => return TLS_ERROR,
950    };
951
952    if ex_data.application_protos.is_empty() {
953        return TLS_ERROR;
954    }
955
956    let mut protos = octets::Octets::with_slice(unsafe {
957        slice::from_raw_parts(inp, in_len as usize)
958    });
959
960    while let Ok(proto) = protos.get_bytes_with_u8_length() {
961        let found = ex_data.application_protos.iter().any(|expected| {
962            trace!(
963                "checking peer ALPN {:?} against {:?}",
964                std::str::from_utf8(proto.as_ref()),
965                std::str::from_utf8(expected.as_slice())
966            );
967
968            if expected.len() == proto.len() &&
969                expected.as_slice() == proto.as_ref()
970            {
971                unsafe {
972                    *out = expected.as_slice().as_ptr();
973                    *out_len = expected.len() as u8;
974                }
975
976                return true;
977            }
978
979            false
980        });
981
982        if found {
983            return 0; // SSL_TLSEXT_ERR_OK
984        }
985    }
986
987    TLS_ERROR
988}

990extern "C" fn new_session(ssl: *mut SSL, session: *mut SSL_SESSION) -> c_int {
991    let ex_data = match ExData::from_ssl_ptr(ssl) {
992        Some(v) => v,
993
994        None => return 0,
995    };
996
997    let handshake = Handshake::new(ssl);
998    let peer_params = handshake.quic_transport_params();
999
1000    // Serialize session object into buffer.
1001    let session_bytes = match get_session_bytes(session) {
1002        Ok(v) => v,
1003        Err(_) => return 0,
1004    };
1005
1006    let mut buffer =
1007        Vec::with_capacity(8 + peer_params.len() + 8 + session_bytes.len());
1008
1009    let session_bytes_len = session_bytes.len() as u64;
1010
1011    if buffer.write(&session_bytes_len.to_be_bytes()).is_err() {
1012        std::mem::forget(handshake);
1013        return 0;
1014    }
1015
1016    if buffer.write(&session_bytes).is_err() {
1017        std::mem::forget(handshake);
1018        return 0;
1019    }
1020
1021    let peer_params_len = peer_params.len() as u64;
1022
1023    if buffer.write(&peer_params_len.to_be_bytes()).is_err() {
1024        std::mem::forget(handshake);
1025        return 0;
1026    }
1027
1028    if buffer.write(peer_params).is_err() {
1029        std::mem::forget(handshake);
1030        return 0;
1031    }
1032
1033    *ex_data.session = Some(buffer);
1034
1035    // Prevent handshake from being freed, as we still need it.
1036    std::mem::forget(handshake);
1037
1038    0
1039}
