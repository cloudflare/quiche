211    #[cfg(windows)]
212    fn load_ca_certs(&mut self) -> Result<()> {
213        unsafe {
214            let cstr = ffi::CString::new("Root").map_err(|_| Error::TlsFail)?;
215            let sys_store =
216                windows_sys::Win32::Security::Cryptography::CertOpenSystemStoreA(
217                    0,
218                    cstr.as_ptr() as windows_sys::core::PCSTR,
219                );
220            if sys_store.is_null() {
221                return Err(Error::TlsFail);
222            }
223
224            let ctx_store = SSL_CTX_get_cert_store(self.as_mut_ptr());
225            if ctx_store.is_null() {
226                return Err(Error::TlsFail);
227            }
228
229            let mut ctx_p = windows_sys::Win32::Security::Cryptography::CertEnumCertificatesInStore(
230                sys_store,
231                ptr::null(),
232            );
233
234            while !ctx_p.is_null() {
235                let in_p = (*ctx_p).pbCertEncoded as *const u8;
236
237                let cert = d2i_X509(
238                    ptr::null_mut(),
239                    &in_p,
240                    (*ctx_p).cbCertEncoded as i32,
241                );
242                if !cert.is_null() {
243                    X509_STORE_add_cert(ctx_store, cert);
244                }
245
246                X509_free(cert);
247
248                ctx_p = windows_sys::Win32::Security::Cryptography::CertEnumCertificatesInStore(
249                    sys_store, ctx_p,
250                );
251            }
252
253            // tidy up
254            windows_sys::Win32::Security::Cryptography::CertFreeCertificateContext(ctx_p);
255            windows_sys::Win32::Security::Cryptography::CertCloseStore(
256                sys_store, 0,
257            );
258        }
259
260        Ok(())
261    }

294    pub fn set_alpn(&mut self, v: &[&[u8]]) -> Result<()> {
295        let mut protos: Vec<u8> = Vec::new();
296
297        for proto in v {
298            protos.push(proto.len() as u8);
299            protos.extend_from_slice(proto);
300        }
301
302        // Configure ALPN for servers.
303        unsafe {
304            SSL_CTX_set_alpn_select_cb(
305                self.as_mut_ptr(),
306                Some(select_alpn),
307                ptr::null_mut(),
308            );
309        }
310
311        // Configure ALPN for clients.
312        map_result_zero_is_success(unsafe {
313            SSL_CTX_set_alpn_protos(
314                self.as_mut_ptr(),
315                protos.as_ptr(),
316                protos.len(),
317            )
318        })
319    }

535    pub fn do_handshake(&mut self, ex_data: &mut ExData) -> Result<()> {
536        self.set_ex_data(*QUICHE_EX_DATA_INDEX, ex_data)?;
537        let rc = unsafe { SSL_do_handshake(self.as_mut_ptr()) };
538        self.set_ex_data::<Connection>(*QUICHE_EX_DATA_INDEX, ptr::null())?;
539
540        self.set_transport_error(ex_data, rc);
541        self.map_result_ssl(rc)

544    pub fn process_post_handshake(&mut self, ex_data: &mut ExData) -> Result<()> {
545        // If SSL_provide_quic_data hasn't been called since we last called
546        // SSL_process_quic_post_handshake, then there's nothing to do.
547        if !self.provided_data_outstanding {
548            return Ok(());
549        }
550        self.provided_data_outstanding = false;
551
552        self.set_ex_data(*QUICHE_EX_DATA_INDEX, ex_data)?;
553        let rc = unsafe { SSL_process_quic_post_handshake(self.as_mut_ptr()) };
554        self.set_ex_data::<Connection>(*QUICHE_EX_DATA_INDEX, ptr::null())?;
555
556        self.set_transport_error(ex_data, rc);
557        self.map_result_ssl(rc)
558    }

599    fn map_result_ssl(&mut self, bssl_result: c_int) -> Result<()> {
600        match bssl_result {
601            1 => Ok(()),
602
603            _ => {
604                let ssl_err = self.get_error(bssl_result);
605                match ssl_err {
606                    // SSL_ERROR_SSL
607                    1 => {
608                        log_ssl_error();
609
610                        Err(Error::TlsFail)
611                    },
612
613                    // SSL_ERROR_WANT_READ
614                    2 => Err(Error::Done),
615
616                    // SSL_ERROR_WANT_WRITE
617                    3 => Err(Error::Done),
618
619                    // SSL_ERROR_WANT_X509_LOOKUP
620                    4 => Err(Error::Done),
621
622                    // SSL_ERROR_SYSCALL
623                    5 => Err(Error::TlsFail),
624
625                    // SSL_ERROR_PENDING_SESSION
626                    11 => Err(Error::Done),
627
628                    // SSL_ERROR_PENDING_CERTIFICATE
629                    12 => Err(Error::Done),
630
631                    // SSL_ERROR_WANT_PRIVATE_KEY_OPERATION
632                    13 => Err(Error::Done),
633
634                    // SSL_ERROR_PENDING_TICKET
635                    14 => Err(Error::Done),
636
637                    // SSL_ERROR_EARLY_DATA_REJECTED
638                    15 => {
639                        self.reset_early_data_reject();
640                        Err(Error::Done)
641                    },
642
643                    // SSL_ERROR_WANT_CERTIFICATE_VERIFY
644                    16 => Err(Error::Done),
645
646                    _ => Err(Error::TlsFail),
647                }
648            },
649        }
650    }
651
652    fn set_transport_error(&mut self, ex_data: &mut ExData, bssl_result: c_int) {
653        // SSL_ERROR_SSL
654        if self.get_error(bssl_result) == 1 {
655            // SSL_ERROR_SSL can't be recovered so ensure we set a
656            // local_error so the connection is closed.
657            // See https://www.openssl.org/docs/man1.1.1/man3/SSL_get_error.html
658            if ex_data.local_error.is_none() {
659                *ex_data.local_error = Some(ConnectionError {
660                    is_app: false,
661                    error_code: INTERNAL_ERROR,
662                    reason: Vec::new(),
663                })
664            }
665        }
666    }

740fn get_cipher_from_ptr(cipher: *const SSL_CIPHER) -> Result<crypto::Algorithm> {
741    let cipher_id = unsafe { SSL_CIPHER_get_id(cipher) };
742
743    let alg = match cipher_id {
744        0x0300_1301 => crypto::Algorithm::AES128_GCM,
745        0x0300_1302 => crypto::Algorithm::AES256_GCM,
746        0x0300_1303 => crypto::Algorithm::ChaCha20_Poly1305,
747        _ => return Err(Error::TlsFail),
748    };
749
750    Ok(alg)
751}

753extern "C" fn set_read_secret(
754    ssl: *mut SSL, level: crypto::Level, cipher: *const SSL_CIPHER,
755    secret: *const u8, secret_len: usize,
756) -> c_int {
757    let ex_data = match ExData::from_ssl_ptr(ssl) {
758        Some(v) => v,
759
760        None => return 0,
761    };
762
763    trace!("{} set read secret lvl={:?}", ex_data.trace_id, level);
764
765    let space = match level {
766        crypto::Level::Initial => &mut ex_data.crypto_ctx[packet::Epoch::Initial],
767        crypto::Level::ZeroRTT =>
768            &mut ex_data.crypto_ctx[packet::Epoch::Application],
769        crypto::Level::Handshake =>
770            &mut ex_data.crypto_ctx[packet::Epoch::Handshake],
771        crypto::Level::OneRTT =>
772            &mut ex_data.crypto_ctx[packet::Epoch::Application],
773    };
774
775    let aead = match get_cipher_from_ptr(cipher) {
776        Ok(v) => v,
777
778        Err(_) => return 0,
779    };
780
781    // 0-RTT read secrets are present only on the server.
782    if level != crypto::Level::ZeroRTT || ex_data.is_server {
783        let secret = unsafe { slice::from_raw_parts(secret, secret_len) };
784
785        let open = match crypto::Open::from_secret(aead, secret) {
786            Ok(v) => v,
787
788            Err(_) => return 0,
789        };
790
791        if level == crypto::Level::ZeroRTT {
792            space.crypto_0rtt_open = Some(open);
793            return 1;
794        }
795
796        space.crypto_open = Some(open);
797    }
798
799    1
800}

803    ssl: *mut SSL, level: crypto::Level, cipher: *const SSL_CIPHER,
804    secret: *const u8, secret_len: usize,
805) -> c_int {
806    let ex_data = match ExData::from_ssl_ptr(ssl) {
807        Some(v) => v,
808
809        None => return 0,
810    };
811
812    trace!("{} set write secret lvl={:?}", ex_data.trace_id, level);
813
814    let space = match level {
815        crypto::Level::Initial => &mut ex_data.crypto_ctx[packet::Epoch::Initial],
816        crypto::Level::ZeroRTT =>
817            &mut ex_data.crypto_ctx[packet::Epoch::Application],

819            &mut ex_data.crypto_ctx[packet::Epoch::Handshake],
820        crypto::Level::OneRTT =>
821            &mut ex_data.crypto_ctx[packet::Epoch::Application],
822    };
823

835            Ok(v) => v,
836
837            Err(_) => return 0,
838        };
839
840        space.crypto_seal = Some(seal);
841    }
842

844}

846extern "C" fn add_handshake_data(
847    ssl: *mut SSL, level: crypto::Level, data: *const u8, len: usize,
848) -> c_int {
849    let ex_data = match ExData::from_ssl_ptr(ssl) {
850        Some(v) => v,
851
852        None => return 0,
853    };
854
855    trace!(
856        "{} write message lvl={:?} len={}",
857        ex_data.trace_id,
858        level,
859        len
860    );
861
862    let buf = unsafe { slice::from_raw_parts(data, len) };
863
864    let space = match level {
865        crypto::Level::Initial => &mut ex_data.crypto_ctx[packet::Epoch::Initial],
866        crypto::Level::ZeroRTT => unreachable!(),
867        crypto::Level::Handshake =>
868            &mut ex_data.crypto_ctx[packet::Epoch::Handshake],
869        crypto::Level::OneRTT =>
870            &mut ex_data.crypto_ctx[packet::Epoch::Application],
871    };
872
873    if space.crypto_stream.send.write(buf, false).is_err() {
874        return 0;
875    }
876
877    1
878}

913extern "C" fn keylog(ssl: *const SSL, line: *const c_char) {
914    let ex_data = match ExData::from_ssl_ptr(ssl) {
915        Some(v) => v,
916
917        None => return,
918    };
919
920    if let Some(keylog) = &mut ex_data.keylog {
921        let data = unsafe { ffi::CStr::from_ptr(line).to_bytes() };
922
923        let mut full_line = Vec::with_capacity(data.len() + 1);
924        full_line.extend_from_slice(data);
925        full_line.push(b'\n');
926
927        keylog.write_all(&full_line[..]).ok();
928        keylog.flush().ok();
929    }
930}

932extern "C" fn select_alpn(
933    ssl: *mut SSL, out: *mut *const u8, out_len: *mut u8, inp: *mut u8,
934    in_len: c_uint, _arg: *mut c_void,
935) -> c_int {
936    // SSL_TLSEXT_ERR_OK 0
937    // SSL_TLSEXT_ERR_ALERT_WARNING 1
938    // SSL_TLSEXT_ERR_ALERT_FATAL 2
939    // SSL_TLSEXT_ERR_NOACK 3
940
941    // Boringssl internally overwrite the return value from this callback, if the
942    // returned value is SSL_TLSEXT_ERR_NOACK and is quic, then the value gets
943    // overwritten to SSL_TLSEXT_ERR_ALERT_FATAL. In contrast openssl/quictls does
944    // not do that, so we need to explicitly respond with
945    // SSL_TLSEXT_ERR_ALERT_FATAL in case it is needed.
946    // TLS_ERROR is redefined for each vendor.
947    let ex_data = match ExData::from_ssl_ptr(ssl) {
948        Some(v) => v,
949
950        None => return TLS_ERROR,
951    };
952
953    if ex_data.application_protos.is_empty() {
954        return TLS_ERROR;
955    }
956
957    let mut protos = octets::Octets::with_slice(unsafe {
958        slice::from_raw_parts(inp, in_len as usize)
959    });
960
961    while let Ok(proto) = protos.get_bytes_with_u8_length() {
962        let found = ex_data.application_protos.iter().any(|expected| {
963            trace!(
964                "checking peer ALPN {:?} against {:?}",
965                std::str::from_utf8(proto.as_ref()),
966                std::str::from_utf8(expected.as_slice())
967            );
968
969            if expected.len() == proto.len() &&
970                expected.as_slice() == proto.as_ref()
971            {
972                unsafe {
973                    *out = expected.as_slice().as_ptr();
974                    *out_len = expected.len() as u8;
975                }
976
977                return true;
978            }
979
980            false
981        });
982
983        if found {
984            return 0; // SSL_TLSEXT_ERR_OK
985        }
986    }
987
988    TLS_ERROR
989}

991extern "C" fn new_session(ssl: *mut SSL, session: *mut SSL_SESSION) -> c_int {
992    let ex_data = match ExData::from_ssl_ptr(ssl) {
993        Some(v) => v,
994
995        None => return 0,
996    };
997
998    let handshake = Handshake::new(ssl);
999    let peer_params = handshake.quic_transport_params();
1000
1001    // Serialize session object into buffer.
1002    let session_bytes = match get_session_bytes(session) {
1003        Ok(v) => v,
1004        Err(_) => return 0,
1005    };
1006
1007    let mut buffer =
1008        Vec::with_capacity(8 + peer_params.len() + 8 + session_bytes.len());
1009
1010    let session_bytes_len = session_bytes.len() as u64;
1011
1012    if buffer.write(&session_bytes_len.to_be_bytes()).is_err() {
1013        std::mem::forget(handshake);
1014        return 0;
1015    }
1016
1017    if buffer.write(&session_bytes).is_err() {
1018        std::mem::forget(handshake);
1019        return 0;
1020    }
1021
1022    let peer_params_len = peer_params.len() as u64;
1023
1024    if buffer.write(&peer_params_len.to_be_bytes()).is_err() {
1025        std::mem::forget(handshake);
1026        return 0;
1027    }
1028
1029    if buffer.write(peer_params).is_err() {
1030        std::mem::forget(handshake);
1031        return 0;
1032    }
1033
1034    *ex_data.session = Some(buffer);
1035
1036    // Prevent handshake from being freed, as we still need it.
1037    std::mem::forget(handshake);
1038
1039    0
1040}
