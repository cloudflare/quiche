27//! Delivery rate estimation.
28//!
29//! This implements the algorithm for estimating delivery rate as described in
30//! <https://tools.ietf.org/html/draft-cheng-iccrg-delivery-rate-estimation-01>
31
32use std::time::Duration;
33use std::time::Instant;

35use crate::recovery::bandwidth::Bandwidth;

38use super::Sent;

40#[derive(Debug)]
41pub struct Rate {
42    delivered: usize,
43
44    delivered_time: Instant,
45
46    first_sent_time: Instant,
47
48    // Packet number of the last sent packet with app limited.
49    end_of_app_limited: u64,
50
51    // Packet number of the last sent packet.
52    last_sent_packet: u64,
53
54    // Packet number of the largest acked packet.
55    largest_acked: u64,
56
57    // Sample of rate estimation.
58    rate_sample: RateSample,
59}
60
61impl Default for Rate {
62    fn default() -> Self {
63        let now = Instant::now();
64
65        Rate {
66            delivered: 0,
67
68            delivered_time: now,
69
70            first_sent_time: now,
71
72            end_of_app_limited: 0,
73
74            last_sent_packet: 0,
75
76            largest_acked: 0,
77
78            rate_sample: RateSample::new(),
79        }
80    }
81}
82
83impl Rate {
84    pub fn on_packet_sent(
85        &mut self, pkt: &mut Sent, bytes_in_flight: usize, bytes_lost: u64,
86    ) {
87        // No packets in flight.
88        if bytes_in_flight == 0 {
89            self.first_sent_time = pkt.time_sent;
90            self.delivered_time = pkt.time_sent;
91        }
92
93        pkt.first_sent_time = self.first_sent_time;
94        pkt.delivered_time = self.delivered_time;
95        pkt.delivered = self.delivered;
96        pkt.is_app_limited = self.app_limited();
97        pkt.tx_in_flight = bytes_in_flight;
98        pkt.lost = bytes_lost;
99
100        self.last_sent_packet = pkt.pkt_num;
101    }
102
103    // Update the delivery rate sample when a packet is acked.
104    pub fn update_rate_sample(&mut self, pkt: &Acked, now: Instant) {
105        self.delivered += pkt.size;
106        self.delivered_time = now;
107
108        // Update info using the newest packet. If rate_sample is not yet
109        // initialized, initialize with the first packet.
110        if self.rate_sample.prior_time.is_none() ||
111            pkt.delivered >= self.rate_sample.prior_delivered
112        {
113            self.rate_sample.prior_delivered = pkt.delivered;
114            self.rate_sample.prior_time = Some(pkt.delivered_time);
115            self.rate_sample.is_app_limited = pkt.is_app_limited;
116            self.rate_sample.send_elapsed =
117                pkt.time_sent.saturating_duration_since(pkt.first_sent_time);
118            self.rate_sample.rtt = pkt.rtt;
119            self.rate_sample.ack_elapsed = self
120                .delivered_time
121                .saturating_duration_since(pkt.delivered_time);
122
123            self.first_sent_time = pkt.time_sent;
124        }
125
126        self.largest_acked = self.largest_acked.max(pkt.pkt_num);
127    }
128
129    pub fn generate_rate_sample(&mut self, min_rtt: Duration) {
130        // End app-limited phase if bubble is ACKed and gone.
131        if self.app_limited() && self.largest_acked > self.end_of_app_limited {
132            self.update_app_limited(false);
133        }
134
135        if self.rate_sample.prior_time.is_some() {
136            let interval = self
137                .rate_sample
138                .send_elapsed
139                .max(self.rate_sample.ack_elapsed);
140
141            self.rate_sample.delivered =
142                self.delivered - self.rate_sample.prior_delivered;
143            self.rate_sample.interval = interval;
144
145            if interval < min_rtt {
146                self.rate_sample.interval = Duration::ZERO;
147
148                // No reliable sample.
149                return;
150            }
151
152            if !interval.is_zero() {
153                let rate_sample_bandwidth = {
154                    let rate_sample_bytes_per_second = (self.rate_sample.delivered
155                        as f64 /
156                        interval.as_secs_f64())
157                        as u64;
158
159                    Bandwidth::from_bytes_per_second(rate_sample_bytes_per_second)
160                };
161
162                // Match the [linux] implementation and only generate a new
163                // sample delivery rate if either:
164                // - the sample was not app_limited
165                // - the new rate is higher than the previous value
166                //
167                // [linux] https://github.com/torvalds/linux/commit/eb8329e0a04db0061f714f033b4454326ba147f4
168                if !self.rate_sample.is_app_limited ||
169                    rate_sample_bandwidth > self.rate_sample.bandwidth
170                {
171                    self.update_delivery_rate(rate_sample_bandwidth);
172                }
173            }
174        }
175    }
176
177    fn update_delivery_rate(&mut self, bandwidth: Bandwidth) {
178        self.rate_sample.bandwidth = bandwidth;
179    }
180
181    pub fn update_app_limited(&mut self, v: bool) {
182        self.end_of_app_limited =
183            if v { self.last_sent_packet.max(1) } else { 0 };
184    }
185
186    pub fn app_limited(&mut self) -> bool {
187        self.end_of_app_limited != 0
188    }
189
190    pub fn delivered(&self) -> usize {
191        self.delivered
192    }
193
194    pub fn sample_delivery_rate(&self) -> Bandwidth {
195        self.rate_sample.bandwidth
196    }
197
198    pub fn sample_rtt(&self) -> Duration {
199        self.rate_sample.rtt
200    }
201
202    pub fn sample_is_app_limited(&self) -> bool {
203        self.rate_sample.is_app_limited
204    }
205
206    pub fn sample_delivered(&self) -> usize {
207        self.rate_sample.delivered
208    }
209
210    pub fn sample_prior_delivered(&self) -> usize {
211        self.rate_sample.prior_delivered
212    }
213}

215#[derive(Debug)]
216struct RateSample {
217    // The sample delivery_rate in bytes/sec
218    bandwidth: Bandwidth,
219
220    is_app_limited: bool,
221
222    interval: Duration,
223
224    delivered: usize,
225
226    prior_delivered: usize,
227
228    prior_time: Option<Instant>,
229
230    send_elapsed: Duration,
231
232    ack_elapsed: Duration,
233
234    rtt: Duration,
235}
236
237impl RateSample {
238    const fn new() -> Self {
239        RateSample {
240            bandwidth: Bandwidth::zero(),
241            is_app_limited: false,
242            interval: Duration::ZERO,
243            delivered: 0,
244            prior_delivered: 0,
245            prior_time: None,
246            send_elapsed: Duration::ZERO,
247            ack_elapsed: Duration::ZERO,
248            rtt: Duration::ZERO,
249        }
250    }

267    // A [RateSample](delivery_rate::RateSample) is app_limited if it was
268    // generated when the [Rate](delivery_rate::Rate) was app_limited.
269    //
270    // The following test generates RateSamples before and after Rate is
271    // app_limited and asserts on app_limited status for samples.
272    #[test]
273    fn sample_is_app_limited() {
274        let config = Config::new(0xbabababa).unwrap();
275        let mut r = LegacyRecovery::new(&config);
276        let mut now = Instant::now();
277        let mss = r.max_datagram_size();
278
279        // Not App Limited prior to any activity
280        assert!(!r.congestion.delivery_rate.app_limited());
281        assert_eq!(r.congestion.delivery_rate.end_of_app_limited, 0);
282        assert!(!r.congestion.delivery_rate.sample_is_app_limited());
283
284        // Send/Ack first batch to generate a new delivery_rate sample.
285        let rtt = Duration::from_secs(2);
286        helper_send_and_ack_packets(&mut r, 0..4, now, rtt, mss);
287
288        // Marking Rate as app_limited.
289        r.delivery_rate_update_app_limited(true);
290        assert!(r.congestion.delivery_rate.app_limited());
291        assert_eq!(r.congestion.delivery_rate.end_of_app_limited, 3);
292
293        // Rate is app_limited.
294        assert!(r.congestion.delivery_rate.app_limited());
295        assert!(!r.congestion.delivery_rate.sample_is_app_limited());
296
297        // Send/Ack second batch to generate a new delivery_rate sample.
298        now += rtt;
299        helper_send_and_ack_packets(&mut r, 4..8, now, rtt, mss);
300
301        // Rate is no longer app limited since we sent a packet larger than
302        // `end_of_app_limited`
303        assert!(!r.congestion.delivery_rate.app_limited());
304        assert_eq!(r.congestion.delivery_rate.end_of_app_limited, 0);
305        // The RateSample is also app_limited
306        assert!(r.congestion.delivery_rate.sample_is_app_limited());
307    }

309    // A [RateSample](delivery_rate::RateSample) is is only updated if either not
310    // app_limited or greater than the previous value.
311    #[test]
312    fn app_limited_delivery_rate() {
313        // confirm that rate sample is not generated when app limited
314        let config = Config::new(0xbabababa).unwrap();
315        let mut r = LegacyRecovery::new(&config);
316        let mut now = Instant::now();
317        let mss = r.max_datagram_size();
318
319        // Not App Limited prior to any activity
320        assert!(!r.congestion.delivery_rate.app_limited());
321        assert_eq!(r.congestion.delivery_rate.end_of_app_limited, 0);
322        assert!(!r.congestion.delivery_rate.sample_is_app_limited());
323
324        // First batch
325        // Send/Ack first batch to generate a new delivery_rate sample.
326        let mut rtt = Duration::from_secs(2);
327        helper_send_and_ack_packets(&mut r, 0..2, now, rtt, mss);
328
329        // Marking Rate as app_limited.
330        r.delivery_rate_update_app_limited(true);
331        assert!(r.congestion.delivery_rate.app_limited());
332        assert_eq!(r.congestion.delivery_rate.end_of_app_limited, 1);
333        assert!(!r.congestion.delivery_rate.sample_is_app_limited());
334
335        let first_delivery_rate = r.delivery_rate().to_bytes_per_second();
336        let expected_delivery_rate = (mss * 2) as u64 / rtt.as_secs();
337        assert_eq!(expected_delivery_rate, 1200);
338        assert_eq!(first_delivery_rate, expected_delivery_rate);
339
340        // Second batch
341        // Since Rtt is larger, the delivery_rate will be smaller and not generate
342        // a new RateSample.
343        now += rtt;
344        rtt = Duration::from_secs(4);
345        helper_send_and_ack_packets(&mut r, 2..4, now, rtt, mss);
346
347        // Rate is no longer app limited since we sent a packet larger than
348        // `end_of_app_limited`
349        assert!(!r.congestion.delivery_rate.app_limited());
350        assert_eq!(r.congestion.delivery_rate.end_of_app_limited, 0);
351        // The RateSample is also app_limited
352        assert!(r.congestion.delivery_rate.sample_is_app_limited());
353
354        // Delivery rate NOT updated since the delivery rate is less than previous
355        // value
356        let expected_delivery_rate = (mss * 2) as u64 / rtt.as_secs();
357        assert_eq!(expected_delivery_rate, 600);
358        let app_limited_delivery_rate = r.delivery_rate().to_bytes_per_second();
359        assert_eq!(app_limited_delivery_rate, first_delivery_rate);
360
361        // Marking Rate as app_limited.
362        r.delivery_rate_update_app_limited(true);
363        assert!(r.congestion.delivery_rate.app_limited());
364        assert_eq!(r.congestion.delivery_rate.end_of_app_limited, 3);
365        // The RateSample is also app_limited
366        assert!(r.congestion.delivery_rate.sample_is_app_limited());
367
368        // Third batch
369        // Since Rtt is smaller, the delivery_rate will be larger and not generate
370        // a new RateSample even when app_limited.
371        now += rtt;
372        rtt = Duration::from_secs(1);
373        helper_send_and_ack_packets(&mut r, 4..6, now, rtt, mss);
374
375        // Rate is no longer app limited since we sent a packet larger than
376        // `end_of_app_limited`
377        assert!(!r.congestion.delivery_rate.app_limited());
378        assert_eq!(r.congestion.delivery_rate.end_of_app_limited, 0);
379        // The RateSample is also app_limited
380        assert!(r.congestion.delivery_rate.sample_is_app_limited());
381
382        // Delivery rate NOT updated since the delivery rate is less than previous
383        // value
384        let expected_delivery_rate = (mss * 2) as u64 / rtt.as_secs();
385        assert_eq!(expected_delivery_rate, 2400);
386        let app_limited_delivery_rate = r.delivery_rate().to_bytes_per_second();
387        assert_eq!(app_limited_delivery_rate, expected_delivery_rate);
388    }

390    #[test]
391    fn rate_check() {
392        let config = Config::new(0xbabababa).unwrap();
393        let mut r = LegacyRecovery::new(&config);
394
395        let now = Instant::now();
396        let mss = r.max_datagram_size();
397
398        // Send 2 packets.
399        for pn in 0..2 {
400            let pkt = test_utils::helper_packet_sent(pn, now, mss);
401
402            r.on_packet_sent(
403                pkt,
404                packet::Epoch::Application,
405                HandshakeStatus::default(),
406                now,
407                "",
408            );
409        }
410
411        let rtt = Duration::from_millis(50);
412        let now = now + rtt;
413
414        // Ack 2 packets.
415        for pn in 0..2 {
416            let acked = Acked {
417                pkt_num: pn,
418                time_sent: now,
419                size: mss,
420                rtt,
421                delivered: 0,
422                delivered_time: now,
423                first_sent_time: now.checked_sub(rtt).unwrap(),
424                is_app_limited: false,
425            };
426
427            r.congestion.delivery_rate.update_rate_sample(&acked, now);
428        }
429
430        // Update rate sample after 1 rtt.
431        r.congestion.delivery_rate.generate_rate_sample(rtt);
432
433        // Bytes acked so far.
434        assert_eq!(r.congestion.delivery_rate.delivered(), 2400);
435
436        // Estimated delivery rate = (1200 x 2) / 0.05s = 48000.
437        assert_eq!(r.delivery_rate().to_bytes_per_second(), 48000);
438    }

441    fn app_limited_cwnd_full() {
442        let config = Config::new(0xbabababa).unwrap();
443        let mut r = LegacyRecovery::new(&config);
444
445        let now = Instant::now();
446        let mss = r.max_datagram_size();
447
448        // Not App Limited prior to any activity
449        assert!(!r.app_limited());
450        assert!(!r.congestion.delivery_rate.sample_is_app_limited());
451
452        // Send 10 packets to fill cwnd.
453        for pn in 0..5 {
454            let pkt = test_utils::helper_packet_sent(pn, now, mss);
455            r.on_packet_sent(
456                pkt,
457                packet::Epoch::Application,
458                HandshakeStatus::default(),
459                now,
460                "",
461            );
462        }
463
464        // App Limited after sending partial cwnd worth of data
465        assert!(r.app_limited());
466        assert!(!r.congestion.delivery_rate.sample_is_app_limited());
467
468        for pn in 5..10 {
469            let pkt = test_utils::helper_packet_sent(pn, now, mss);
470            r.on_packet_sent(
471                pkt,
472                packet::Epoch::Application,
473                HandshakeStatus::default(),
474                now,
475                "",
476            );
477        }
478
479        // Not App Limited after sending full cwnd worth of data
480        assert!(!r.app_limited());
481        assert!(!r.congestion.delivery_rate.sample_is_app_limited());
482    }

484    fn helper_send_and_ack_packets(
485        recovery: &mut LegacyRecovery, range: Range<u64>, now: Instant,
486        rtt: Duration, mss: usize,
487    ) {
488        for pn in range.clone() {
489            let pkt = test_utils::helper_packet_sent(pn, now, mss);
490            recovery.on_packet_sent(
491                pkt,
492                packet::Epoch::Application,
493                HandshakeStatus::default(),
494                now,
495                "",
496            );
497        }
498
499        let packet_count = range.clone().count();
500
501        // Ack packets, which generates a new delivery_rate
502        let mut acked = ranges::RangeSet::default();
503        acked.insert(range);
504
505        let ack_outcome = recovery
506            .on_ack_received(
507                &acked,
508                25,
509                packet::Epoch::Application,
510                HandshakeStatus::default(),
511                now + rtt,
512                None,
513                "",
514            )
515            .unwrap();
516
517        assert_eq!(ack_outcome, OnAckReceivedOutcome {
518            lost_packets: 0,
519            lost_bytes: 0,
520            acked_bytes: mss * packet_count,
521            spurious_losses: 0,
522        });
523    }
