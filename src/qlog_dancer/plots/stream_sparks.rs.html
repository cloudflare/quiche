306    fn from_data_store(ds: &Datastore, stream_id: u64) -> Option<Self> {
307        let is_request = match ds.application_proto {
308            ApplicationProto::Http3 => stream_id.is_multiple_of(4),
309            ApplicationProto::Http2 => true,
310        };
311
312        if is_request {
313            if let Some(req) = ds.http_requests.get(&stream_id) {
314                match ds.vantage_point {
315                    VantagePoint::Client =>
316                        Some(SparkCaption::request_at_client_from_stub(
317                            ds.application_proto,
318                            req,
319                        )),
320                    VantagePoint::Server => Some(
321                        SparkCaption::request_at_server_from_stub(stream_id, req),
322                    ),
323                }
324            } else {
325                None
326            }
327        } else {
328            let summary = format!("ID= {}", stream_id);
329            Some(SparkCaption::UniStream { summary })
330        }
331    }

515fn plot_legend<DB: DrawingBackend>(
516    color: &RGBColor, root: &DrawingArea<DB, plotters::coord::Shift>, x: i32,
517    y: i32, params: &SparkPlotsParams, transmission_type: TransmissionType,
518) {
519    // TODO: only considers vert layout, needs to consider grid layout too
520    let style = &("monospace", 12).into_text_style(root).color(color);
521    let x: i32 = x + params.spark_dimension_x as i32 + 300;
522    let y: i32 = y;
523    let line_len: i32 = 50;
524    let nl_off: i32 = 20;
525
526    root.draw_text("Legend", style, (x, y)).unwrap();
527
528    // stream data
529    let label = match transmission_type {
530        TransmissionType::Download => "Stream data read",
531        TransmissionType::Upload => "Stream data sent",
532    };
533    root.draw(&PathElement::new(
534        vec![(x, y + nl_off), (x + line_len, y + nl_off)],
535        PURPLE_500,
536    ))
537    .unwrap();
538    root.draw_text(label, style, (x + line_len + 10, y + (nl_off) - 5))
539        .unwrap();
540
541    // stream data buffered
542    root.draw(&PathElement::new(
543        vec![(x, y + nl_off * 2), (x + line_len, y + nl_off * 2)],
544        MAGENTA,
545    ))
546    .unwrap();
547    root.draw_text(
548        "Stream data buffered",
549        style,
550        (x + line_len + 10, y + (nl_off * 2) - 5),
551    )
552    .unwrap();
553
554    // max stream data
555    let label = match transmission_type {
556        TransmissionType::Download => "Max stream data sent",
557        TransmissionType::Upload => "Max stream data received",
558    };
559
560    root.draw(&PathElement::new(
561        vec![(x, y + nl_off * 3), (x + line_len, y + nl_off * 3)],
562        MUSTARD,
563    ))
564    .unwrap();
565    root.draw_text(label, style, (x + line_len + 10, y + (nl_off * 3) - 5))
566        .unwrap();
567
568    // headers transmitted
569    root.draw(&PathElement::new(
570        vec![(x, y + nl_off * 4), (x + line_len, y + nl_off * 4)],
571        CYAN,
572    ))
573    .unwrap();
574    root.draw_text(
575        "Request headers sent",
576        style,
577        (x + line_len + 10, y + (nl_off * 4) - 5),
578    )
579    .unwrap();
580
581    // first data frame
582    root.draw(&PathElement::new(
583        vec![(x, y + nl_off * 5), (x + line_len, y + nl_off * 5)],
584        ORANGE,
585    ))
586    .unwrap();
587    root.draw_text(
588        "First data frame read",
589        style,
590        (x + line_len + 10, y + (nl_off * 5) - 5),
591    )
592    .unwrap();
593
594    // last data frame
595    root.draw(&PathElement::new(
596        vec![(x, y + nl_off * 6), (x + line_len, y + nl_off * 6)],
597        BROWN,
598    ))
599    .unwrap();
600    root.draw_text(
601        "Last data frame read",
602        style,
603        (x + line_len + 10, y + (nl_off * 6) - 5),
604    )
605    .unwrap();
606}

608#[allow(clippy::too_many_arguments)]
609pub fn plot_sparks(
610    params: &SparkPlotsParams, filename: &str, ss: &SeriesStore, ds: &Datastore,
611    abs_dl_ty: &ChartOutputType, rel_dl_ty: &ChartOutputType,
612    abs_ul_ty: &ChartOutputType, rel_ul_ty: &ChartOutputType,
613) {
614    let (stream_frame_dl_series_to_plot, stream_frame_ul_series_to_plot) =
615        match (ds.vantage_point, ds.application_proto) {
616            (VantagePoint::Client, ApplicationProto::Http3) => (
617                &ss.received_stream_frames_series,
618                &ss.sent_stream_frames_series,
619            ),
620            (VantagePoint::Server, ApplicationProto::Http3) => (
621                &ss.sent_stream_frames_series,
622                &ss.received_stream_frames_series,
623            ),
624            (VantagePoint::Client, ApplicationProto::Http2) =>
625                (&ss.received_data_frames_series, &ss.sent_data_frames_series),
626            _ => unimplemented!(),
627        };
628
629    #[cfg(not(target_arch = "wasm32"))]
630    let (total_dl_sparks, total_ul_sparks) =
631        if params.bidi_only && ds.application_proto == ApplicationProto::Http3 {
632            let total_dl = stream_frame_dl_series_to_plot
633                .iter()
634                .filter(|(k, _)| (*k & 0x2) == 0)
635                .count() as u32;
636
637            let total_ul = stream_frame_ul_series_to_plot
638                .iter()
639                .filter(|(k, _)| (*k & 0x2) == 0)
640                .count() as u32;
641
642            (total_dl, total_ul)
643        } else {
644            (
645                stream_frame_dl_series_to_plot.len() as u32,
646                stream_frame_ul_series_to_plot.len() as u32,
647            )
648        };
649
650    // TODO: replace with div_ceil once stablized- https://github.com/rust-lang/rust/issues/88581
651    #[cfg(not(target_arch = "wasm32"))]
652    let total_dl_rows: u32 = total_dl_sparks / params.sparks_per_row +
653        u32::from(total_dl_sparks % params.sparks_per_row != 0);
654    #[cfg(not(target_arch = "wasm32"))]
655    let total_ul_rows: u32 = total_ul_sparks / params.sparks_per_row +
656        u32::from(total_ul_sparks % params.sparks_per_row != 0);
657
658    #[cfg(not(target_arch = "wasm32"))]
659    let caption_area_width = if params.captions_on_top {
660        0
661    } else {
662        params.caption_area_width
663    };
664
665    let caption_area_height = if params.captions_on_top {
666        params.caption_area_height
667    } else {
668        0
669    };
670
671    #[cfg(not(target_arch = "wasm32"))]
672    let spark_and_captions_width = params.spark_dimension_x +
673        (params.spark_offset_x as f32 * 1.2) as u32 +
674        caption_area_width;
675    #[cfg(not(target_arch = "wasm32"))]
676    let spark_and_captions_height = params.spark_dimension_y +
677        params.spark_offset_y +
678        (params.label_area_height) +
679        caption_area_height;
680
681    #[cfg(not(target_arch = "wasm32"))]
682    let root_width: u32 = params.sparks_per_row * spark_and_captions_width;
683
684    #[cfg(not(target_arch = "wasm32"))]
685    let dl_root_height: u32 = total_dl_rows * spark_and_captions_height;
686
687    #[cfg(not(target_arch = "wasm32"))]
688    let ul_root_height: u32 = total_ul_rows * spark_and_captions_height;
689
690    let mut x_min = match ds.vantage_point {
691        VantagePoint::Client => ss.received_x_min,
692        VantagePoint::Server => ss.sent_x_min,
693    };
694
695    let mut x_max = match ds.vantage_point {
696        VantagePoint::Client => ss.received_x_max,
697        VantagePoint::Server => ss.sent_x_max,
698    };
699
700    // Clamp plot x-axis if the user told us.
701    if let Some(s) = params.clamp.start {
702        x_min = s;
703    }
704
705    if let Some(e) = params.clamp.end {
706        x_max = x_max.min(e);
707    }
708
709    let abs_dl_chart_config = ChartConfig {
710        title: "stream-spark-dl-absolute".into(),
711        input_filename: filename.into(),
712        clamp: params.clamp.clone(),
713        app_proto: ds.application_proto,
714        host: ds.host.clone(),
715        session_id: ds.session_id,
716        ty: abs_dl_ty.clone(),
717    };
718    abs_dl_chart_config.init_chart_dir();
719
720    let rel_dl_chart_config = ChartConfig {
721        title: "stream-spark-dl-relative".into(),
722        input_filename: filename.into(),
723        clamp: params.clamp.clone(),
724        app_proto: ds.application_proto,

1124            y_max,

1148#[allow(clippy::too_many_arguments)]
1149fn draw_captions<DB: DrawingBackend>(
1150    abs_dl_root: &DrawingArea<DB, plotters::coord::Shift>,
1151    rel_dl_root: &DrawingArea<DB, plotters::coord::Shift>,
1152    abs_ul_root: &DrawingArea<DB, plotters::coord::Shift>,
1153    rel_ul_root: &DrawingArea<DB, plotters::coord::Shift>,
1154    params: &SparkPlotsParams, ds: &Datastore, stream_id: u64, upload_data: bool,
1155    upper_x: u32, dl_upper_y: u32, ul_upper_y: u32,
1156) {
1157    if let Some(captions) = SparkCaption::from_data_store(ds, stream_id) {
1158        let x_offset = if params.captions_on_top {
1159            0
1160        } else if ds.application_proto == ApplicationProto::Http2 {
1161            params.spark_dimension_x + 30
1162        } else {
1163            params.spark_dimension_x
1164        };
1165
1166        captions.draw(
1167            &params.colors.caption,
1168            abs_dl_root,
1169            (upper_x + x_offset) as i32,
1170            dl_upper_y as i32,
1171        );
1172        captions.draw(
1173            &params.colors.caption,
1174            rel_dl_root,
1175            (upper_x + x_offset) as i32,
1176            dl_upper_y as i32,
1177        );
1178
1179        if upload_data {
1180            captions.draw(
1181                &params.colors.caption,
1182                abs_ul_root,
1183                (upper_x + x_offset) as i32,
1184                ul_upper_y as i32,
1185            );
1186            captions.draw(
1187                &params.colors.caption,
1188                rel_ul_root,
1189                (upper_x + x_offset) as i32,
1190                ul_upper_y as i32,
1191            );
1192        }
1193    }
1194}

1216fn draw_buffered_data_line<DB: DrawingBackend>(

1233            .unwrap();
1234
1235        rel_chart
1236            .draw_series(
1237                buffered_data
1238                    .iter()
1239                    .map(|point| TriangleMarker::new(*point, 3, MAGENTA)),
1240            )
1241            .unwrap();

1245fn draw_stream_max_line<DB: DrawingBackend>(
1246    stream_id: u64, ss: &SeriesStore, vantage_point: &VantagePoint,

1252        (VantagePoint::Client, TransmissionType::Download) =>

1254        (VantagePoint::Client, TransmissionType::Upload) =>
1255            ss.received_stream_max_data.get(&stream_id),
1256        (VantagePoint::Server, TransmissionType::Download) =>
1257            ss.received_stream_max_data.get(&stream_id),
1258        (VantagePoint::Server, TransmissionType::Upload) =>
1259            ss.sent_stream_max_data.get(&stream_id),
1260    };
1261
1262    if let Some(stream_max_data) = stream_max_data_to_plot {
1263        abs_chart
1264            .draw_series(LineSeries::new(stream_max_data.clone(), MUSTARD))
1265            .unwrap();
1266
1267        rel_chart
1268            .draw_series(LineSeries::new(stream_max_data.clone(), MUSTARD))
1269            .unwrap();
1270    }

1273fn draw_request_timing_lines<DB: DrawingBackend>(
1274    ds: &Datastore, stream_id: u64, y_max: u64,
1275    abs_chart: &mut ChartContext<DB, Cartesian2d<RangedCoordf32, RangedCoordu64>>,
1276    rel_chart: &mut ChartContext<DB, Cartesian2d<RangedCoordf32, RangedCoordu64>>,
1277) {
1278    if let Some(req) = ds.http_requests.get(&stream_id) {
1279        if let Some(h) = req.time_first_headers_rx {
1280            let points = vec![(h, 0), (h, y_max)];
1281            abs_chart
1282                .draw_series(LineSeries::new(points, GREEN))
1283                .unwrap();
1284
1285            // don't care about rx events on the relative spark chart
1286        }
1287
1288        if let Some(h) = req.time_first_headers_tx {
1289            let points = vec![(h, 0), (h, y_max)];
1290            abs_chart
1291                .draw_series(LineSeries::new(points.clone(), CYAN))
1292                .unwrap();
1293            rel_chart
1294                .draw_series(LineSeries::new(points, CYAN))
1295                .unwrap();
1296        }
1297
1298        let (first_data_to_plot, last_data_to_plot) = match ds.vantage_point {
1299            VantagePoint::Client =>
1300                (req.time_first_data_rx, req.time_last_data_rx),
1301            VantagePoint::Server =>
1302                (req.time_first_data_tx, req.time_last_data_tx),
1303        };
1304
1305        if let Some(t) = first_data_to_plot {
1306            let points = vec![(t, 0), (t, y_max)];
1307            abs_chart
1308                .draw_series(LineSeries::new(points.clone(), ORANGE))
1309                .unwrap();
1310            rel_chart
1311                .draw_series(LineSeries::new(points, ORANGE))
1312                .unwrap();
1313        }
1314
1315        if let Some(t) = last_data_to_plot {
1316            let points = vec![(t, 0), (t, y_max)];
1317            abs_chart
1318                .draw_series(LineSeries::new(points.clone(), BROWN))
1319                .unwrap();
1320            rel_chart
1321                .draw_series(LineSeries::new(points, BROWN))
1322                .unwrap();
1323        }
1324    }
1325}
