27use packet_sent::draw_packet_sent_received_plot;
28use plotters::prelude::*;
29
30use crate::plots::*;
31
32use crate::datastore::Datastore;
33use crate::seriesstore::SeriesStore;
34
35pub fn plot_packet_received(
36    params: &PlotParameters, filename: &str, ss: &SeriesStore, ds: &Datastore,
37    ty: &ChartOutputType,
38) {
39    let chart_config = ChartConfig {
40        title: "packet-received".into(),
41        input_filename: filename.into(),
42        clamp: params.clamp.clone(),
43        app_proto: ds.application_proto,
44        host: ds.host.clone(),
45        session_id: ds.session_id,
46        ty: ty.clone(),
47    };
48
49    chart_config.init_chart_dir();
50
51    #[cfg(not(target_arch = "wasm32"))]
52    let chart_path = chart_config.chart_filepath();
53
54    #[cfg(not(target_arch = "wasm32"))]
55    let root = make_chart_bitmap_area(
56        &chart_path,
57        params.chart_size,
58        params.colors,
59        params.chart_margin,
60    );
61
62    #[cfg(target_arch = "wasm32")]
63    let canvas_id: String = chart_config.canvas_id().unwrap_or_default();
64
65    #[cfg(target_arch = "wasm32")]
66    let root =
67        make_chart_canvas_area(&canvas_id, params.colors, params.chart_margin);
68
69    let (raw_timings, _remainder) = root.split_vertically((60).percent());
70
71    draw_packet_sent_received_plot(false, filename, params, ss, &raw_timings);
72}
