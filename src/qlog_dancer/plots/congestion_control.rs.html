41pub fn draw_congestion_plot<'a, DB: DrawingBackend + 'a>(
42    params: &PlotParameters, axis: &XYMinMax<u64>, ss: &SeriesStore,
43    ds: &Datastore, plot: &plotters::drawing::DrawingArea<DB, Shift>,
44) -> ChartContext<'a, DB, Cartesian2d<RangedCoordf32, RangedCoordu64>> {
45    let mut builder = ChartBuilder::on(plot);
46
47    builder
48        .x_label_area_size(params.area_margin.x)
49        .y_label_area_size(params.area_margin.y);
50
51    if params.display_chart_title {
52        builder
53            .caption("Congestion", chart_subtitle_style(&params.colors.caption));
54    }
55
56    let extended_y =
57        axis.y_range.start..axis.y_range.end + (axis.y_range.end / 10) * 5;
58
59    let mut plot = builder
60        .build_cartesian_2d(axis.x.range(), extended_y.clone())
61        .unwrap();
62
63    draw_mesh(
64        &params.colors,
65        "Relative time (ms)",
66        "Data (bytes)",
67        false,
68        &mut plot,
69    );
70
71    draw_cc_updates(
72        &ds.congestion_state_updates,
73        axis.y_range.clone(),
74        extended_y,
75        &mut plot,
76    );
77    draw_bytes_in_flight(&ss.local_bytes_in_flight, &mut plot);
78    draw_cwnd(&ss.local_cwnd, &mut plot);
79    draw_ssthresh(&ss.local_ssthresh, &mut plot);
80
81    if params.display_legend {
82        plot.configure_series_labels()
83            .label_font(chart_label_style(&params.colors.caption))
84            .background_style(params.colors.fill.mix(0.8))
85            .border_style(params.colors.axis)
86            .position(SeriesLabelPosition::UpperLeft)
87            .draw()
88            .unwrap();
89    }
90
91    plot
92}

94#[cfg(target_arch = "wasm32")]
95pub fn plot_cc_plot<'a>(
96    params: &PlotParameters, ss: &SeriesStore, ds: &Datastore, canvas_id: &str,
97) -> ChartContext<'a, CanvasBackend, Cartesian2d<RangedCoordf32, RangedCoordu64>>
98{
99    let root =
100        make_chart_canvas_area(&canvas_id, params.colors, params.chart_margin);
101
102    let cwnd_y_max = if let Some(y_max) = params.cwnd_y_max {
103        y_max
104    } else {
105        // add a bit of margin
106        ss.y_max_congestion_plot + ss.y_max_congestion_plot / 10
107    };
108
109    // TODO set minimum
110    let axis = super::minmax::XYMinMax::init(
111        ss.sent_x_min..ss.sent_x_max,
112        params.clamp.start,
113        params.clamp.end,
114        0..cwnd_y_max,
115    );
116
117    draw_congestion_plot(params, &axis, ss, ds, &root)
118}

120fn draw_cc_updates<DB: DrawingBackend>(
121    data: &[(f32, u64, String)], y_range: std::ops::Range<u64>,
122    y_range_extended: std::ops::Range<u64>,
123    congestion_chart: &mut ChartContext<
124        DB,
125        Cartesian2d<RangedCoordf32, RangedCoordu64>,
126    >,
127) {
128    let my_label = |x: f32, y: u64, name: &str| {
129        let color = cc_state_to_color(name);
130        let text_width = name.len() as i32 * 6; // Rough estimate
131        let text_height = 12;
132
133        return EmptyElement::at((x, y))
134            //+ Circle::new((0, 0), 1, ShapeStyle::from(color).filled())
135            + Rectangle::new([(1, -2), (1 + text_width, text_height)],
136                        WHITE.mix(0.7).filled()) // 0.7 alpha for transparency
137            + Text::new(
138                name.to_owned(),
139                (1, 0),
140                ("sans-serif", 12.0).into_font().color(&color),
141            );
142    };
143
144    let mut woggle = vec![0];
145
146    for i in 1..5 {
147        woggle.push((y_range.end / 10) * i);
148    }
149
150    // Add labels from right to left, to ensure the z-ordering doesn't add lines
151    // over the labels
152    let iter = data.iter().rev();
153    for cc_state in iter {
154        let color = cc_state_to_color(&cc_state.2);
155        let x = cc_state.0;
156
157        let line_coords =
158            [(x, y_range_extended.start), (x, y_range_extended.end)];
159
160        congestion_chart
161            .draw_series(LineSeries::new(line_coords, color))
162            .unwrap();
163
164        let area = congestion_chart.plotting_area();
165
166        // control the vertical position of the text label to avoid overlaps.
167        let y = y_range.end + woggle[0];
168        area.draw(&my_label(x, y, &cc_state.2)).unwrap();
169
170        woggle.rotate_left(1);
171    }
172}

204// Colors from ColorCycle list
205fn cc_state_to_color(cc_state: &str) -> RGBColor {
206    match cc_state {
207        // Cubic and Reno
208        "slow_start" => RGBColor(204, 81, 81),
209        "recovery" => RGBColor(127, 51, 51),
210        "congestion_avoidance" => RGBColor(81, 204, 204),
211
212        // BBR
213        "bbr_startup" => RGBColor(204, 81, 81),
214        "bbr_drain" => RGBColor(127, 51, 51),
215        "bbr_probe_bw" => RGBColor(81, 204, 204),
216        "bbr_probe_rtt" => RGBColor(51, 127, 127),
217
218        _ => BLACK,
219    }
220}
