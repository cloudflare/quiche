27//! Multiplexing charts
28//! A single image file containing various renderings of stream and H3 DATA
29
30use log::warn;
31use plotters::prelude::*;
32
33use crate::datastore::VantagePoint;
34use crate::plots::colors::*;
35use crate::plots::*;
36
37use crate::datastore::Datastore;
38use crate::seriesstore::SeriesStore;
39
40pub struct MultiplexPlotsParams {
41    pub clamp: ClampParams,
42    pub bidi_only: bool,
43    pub width: u32,
44    pub y_spacer: u32,
45    pub right_shrink_margin: u32,
46    pub combined_plot_height: u32,
47    pub frame_swimlane_height: u32,
48    pub bubbles_swimlane_height: u32,
49    pub colors: PlotColors,
50    pub area_margin: AreaMargin,
51}
52
53impl Default for MultiplexPlotsParams {
54    fn default() -> Self {
55        Self {
56            bidi_only: true,
57            width: 4000,
58            y_spacer: 20,
59            right_shrink_margin: 50,
60            combined_plot_height: 300,
61            frame_swimlane_height: 60,
62            bubbles_swimlane_height: 100,
63            clamp: Default::default(),
64            colors: Default::default(),
65            area_margin: AreaMargin { x: 40, y: 0 },
66        }
67    }
68}
69
70pub fn plot_stream_multiplexing(
71    params: &MultiplexPlotsParams, filename: &str, ss: &SeriesStore,
72    ds: &Datastore, ty: &ChartOutputType,
73) {
74    let (mut x_min, mut x_max) = match ds.vantage_point {
75        VantagePoint::Client => (ss.received_x_min, ss.received_x_max),
76        VantagePoint::Server => (ss.sent_x_min, ss.sent_x_max),
77    };
78
79    // Clamp plot x-axis if the user told us.
80    if let Some(s) = params.clamp.start {
81        x_min = s;
82    }
83
84    if let Some(e) = params.clamp.end {
85        x_max = x_max.min(e);
86    }
87
88    let http_requests = match ds.vantage_point {
89        VantagePoint::Client => &ds.http_requests,
90        VantagePoint::Server => &ds.http_requests,
91    };
92
93    let y_spacer = 50;
94
95    let data_frames_all_combined_x = params.width - params.right_shrink_margin;
96    let data_frames_all_combined_y: i32 = params.combined_plot_height as i32;
97
98    let data_frames_mini_offset_y = data_frames_all_combined_y + y_spacer;
99    let data_frames_mini_x = params.width - params.right_shrink_margin;
100    let data_frames_mini_y = params.frame_swimlane_height;
101
102    let bubbles_x = params.width - params.right_shrink_margin;
103    let bubbles_y = params.bubbles_swimlane_height;
104    let bubbles_offset_y =
105        data_frames_mini_offset_y + (http_requests.len() * 25) as i32;
106
107    let stream_frames_all_combined_x = params.width - params.right_shrink_margin;
108    let stream_frames_all_combined_y = params.combined_plot_height;
109    let stream_frames_all_combined_offset_y =
110        bubbles_offset_y + (http_requests.len() * 70) as i32;
111
112    let stream_frames_mini_offset_y = stream_frames_all_combined_offset_y +
113        stream_frames_all_combined_y as i32 +
114        y_spacer;
115    let stream_frames_mini_x = params.width - params.right_shrink_margin;
116    let stream_frames_mini_y = params.frame_swimlane_height;
117
118    #[cfg(not(target_arch = "wasm32"))]
119    let full_y =
120        stream_frames_mini_offset_y as u32 + (http_requests.len() * 25) as u32;
121
122    let chart_config = ChartConfig {
123        title: "stream-multiplexing".into(),
124        input_filename: filename.into(),
125        clamp: params.clamp.clone(),
126        app_proto: ds.application_proto,
127        host: ds.host.clone(),
128        session_id: ds.session_id,
129        ty: ty.clone(),
130    };
131
132    chart_config.init_chart_dir();
133
134    let margin = ChartMargin {
135        top: 20,
136        bottom: 20,
137        left: 20,
138        right: 20,
139    };
140
141    #[cfg(not(target_arch = "wasm32"))]
142    let size = ChartSize {
143        width: params.width,
144        height: full_y,
145    };
146
147    #[cfg(not(target_arch = "wasm32"))]
148    let chart_path = chart_config.chart_filepath();
149
150    #[cfg(not(target_arch = "wasm32"))]
151    let root = make_chart_bitmap_area(&chart_path, size, params.colors, margin);
152
153    #[cfg(target_arch = "wasm32")]
154    let canvas_id: String = chart_config.canvas_id().unwrap_or_default();
155
156    #[cfg(target_arch = "wasm32")]
157    let root = make_chart_canvas_area(&canvas_id, params.colors, margin);
158
159    let stream_frame_series = match ds.vantage_point {
160        VantagePoint::Client => &ss.received_stream_frames_series,
161        VantagePoint::Server => &ss.sent_stream_frames_series,
162    };
163
164    //// Plot the data frames first
165    let mut color_cyle = ColorCycle::default();
166
167    color_cyle.reset();
168
169    let data_frames_all_combined_temp_root = root.clone().shrink(
170        (0, 0),
171        (data_frames_all_combined_x, data_frames_all_combined_y),
172    );
173
174    let mut data_frames_all_combined_chart =
175        ChartBuilder::on(&data_frames_all_combined_temp_root)
176            .caption(
177                "DATA frame events",
178                chart_title_style(&params.colors.caption),
179            )
180            .x_label_area_size(params.area_margin.x)
181            .y_label_area_size(params.area_margin.y)
182            .build_cartesian_2d(x_min..x_max, 0..2)
183            .unwrap();
184
185    data_frames_all_combined_chart
186        .configure_mesh()
187        .disable_mesh()
188        .axis_style(params.colors.axis)
189        .label_style(chart_label_style(&params.colors.caption))
190        .draw()
191        .unwrap();
192
193    for (id, stub) in http_requests {
194        if params.bidi_only && (id & 0x2) != 0 {
195            continue;
196        }
197
198        let data_frames = match ds.vantage_point {
199            VantagePoint::Client => &stub.time_data_rx_set,
200            VantagePoint::Server => &stub.time_data_tx_set,
201        };
202
203        let style = color_cyle.next_color();
204
205        let lines = data_frames
206            .iter()
207            .map(|point| PathElement::new([(point.0, 2), (point.0, 0)], style));
208
209        data_frames_all_combined_chart
210            .draw_series(lines.clone())
211            .unwrap();
212    }
213
214    //// Now plot the stream frames combined
215    color_cyle.reset();
216
217    let data_frames_all_combined_temp_root = root.clone().shrink(
218        (0, stream_frames_all_combined_offset_y),
219        (stream_frames_all_combined_x, stream_frames_all_combined_y),
220    );
221
222    let mut stream_frames_all_combined_chart =
223        ChartBuilder::on(&data_frames_all_combined_temp_root)
224            .caption(
225                "STREAM frame events",
226                chart_title_style(&params.colors.caption),
227            )
228            .x_label_area_size(params.area_margin.x)
229            .y_label_area_size(params.area_margin.y)
230            .build_cartesian_2d(x_min..x_max, 0..1)
231            .unwrap();
232
233    stream_frames_all_combined_chart
234        .configure_mesh()
235        .disable_mesh()
236        .axis_style(params.colors.axis)
237        .label_style(chart_label_style(&params.colors.caption))
238        .draw()
239        .unwrap();
240
241    for (id, stream_frames) in stream_frame_series {
242        if params.bidi_only && (id & 0x2) != 0 {
243            continue;
244        }
245
246        let style = color_cyle.next_color();
247
248        let lines = stream_frames
249            .iter()
250            .map(|point| PathElement::new([(point.0, 2), (point.0, 0)], style));
251
252        stream_frames_all_combined_chart.draw_series(lines).unwrap();
253    }
254
255    //// now the mini charts
256    color_cyle.reset();
257
258    let mut upper_y = 0;
259    let mut bubbles_upper_y = 0;
260    let mini_chart_axis_y_range = 0..4;
261    let marker_y = 2;
262    let marker_size = 5;
263
264    let max_data_len = match ds.vantage_point {
265        VantagePoint::Client => ds.largest_data_frame_rx_length_global,
266        VantagePoint::Server => ds.largest_data_frame_tx_length_global,
267    };
268
269    if max_data_len == 0 {
270        warn!("max_data_len = 0, skipping remaining charts");
271        return;
272    }
273
274    for (id, stub) in http_requests {
275        if params.bidi_only && (id & 0x2) != 0 {
276            continue;
277        }
278
279        let data_frames = match ds.vantage_point {
280            VantagePoint::Client => &stub.time_data_rx_set,
281            VantagePoint::Server => &stub.time_data_tx_set,
282        };
283
284        let headers_init = match ds.vantage_point {
285            VantagePoint::Client => stub.time_first_headers_tx,
286            VantagePoint::Server => stub.time_first_headers_rx,
287        };
288
289        let headers_actioned = match ds.vantage_point {
290            VantagePoint::Client => stub.time_first_headers_rx,
291            VantagePoint::Server => stub.time_first_headers_tx,
292        };
293
294        let data_frames_temp_root = root.clone().shrink(
295            (0, upper_y + data_frames_mini_offset_y),
296            (data_frames_mini_x, data_frames_mini_y),
297        );
298
299        let stream_frames_temp_root = root.clone().shrink(
300            (0, upper_y + stream_frames_mini_offset_y),
301            (stream_frames_mini_x, stream_frames_mini_y),
302        );
303
304        upper_y += 20;
305
306        let mut data_frames_mini_independent_chart =
307            ChartBuilder::on(&data_frames_temp_root)
308                .x_label_area_size(params.area_margin.x)
309                .y_label_area_size(params.area_margin.y)
310                .build_cartesian_2d(x_min..x_max, mini_chart_axis_y_range.clone())
311                .unwrap();
312
313        data_frames_mini_independent_chart
314            .configure_mesh()
315            .disable_y_mesh()
316            .disable_y_axis()
317            .axis_style(params.colors.axis)
318            .draw()
319            .unwrap();
320
321        // let's jump ahead here and plot the mini chart header events, to avoid
322        // having to loop again later
323        let mut stream_frames_mini_independent_chart =
324            ChartBuilder::on(&stream_frames_temp_root)
325                .x_label_area_size(params.area_margin.x)
326                .y_label_area_size(params.area_margin.y)
327                .build_cartesian_2d(x_min..x_max, mini_chart_axis_y_range.clone())
328                .unwrap();
329
330        stream_frames_mini_independent_chart
331            .configure_mesh()
332            .disable_y_mesh()
333            .disable_y_axis()
334            .axis_style(params.colors.axis)
335            .draw()
336            .unwrap();
337
338        let style = color_cyle.next_color();
339
340        if let Some(h) = headers_init {
341            data_frames_mini_independent_chart
342                .draw_series([TriangleMarker::new(
343                    (h, marker_y),
344                    marker_size,
345                    style,
346                )])
347                .unwrap();
348
349            stream_frames_mini_independent_chart
350                .draw_series([TriangleMarker::new(
351                    (h, marker_y),
352                    marker_size,
353                    style,
354                )])
355                .unwrap();
356        }
357
358        if let Some(h) = headers_actioned {
359            data_frames_mini_independent_chart
360                .draw_series([TriangleMarker::new(
361                    (h, marker_y),
362                    marker_size,
363                    style,
364                )])
365                .unwrap();
366
367            stream_frames_mini_independent_chart
368                .draw_series([TriangleMarker::new(
369                    (h, marker_y),
370                    marker_size,
371                    style,
372                )])
373                .unwrap();
374        }
375
376        let lines = data_frames
377            .iter()
378            .map(|point| PathElement::new([(point.0, 2), (point.0, 0)], style));
379
380        data_frames_mini_independent_chart
381            .draw_series(lines.clone())
382            .unwrap();
383
384        if let Some(stream_frames) = stream_frame_series.get(id) {
385            let lines = stream_frames.iter().map(|point| {
386                PathElement::new([(point.0, 2), (point.0, 0)], style)
387            });
388
389            stream_frames_mini_independent_chart
390                .draw_series(lines.clone())
391                .unwrap();
392        }
393
394        let bubbles_temp_root = root.clone().shrink(
395            (0, bubbles_upper_y + bubbles_offset_y),
396            (bubbles_x, bubbles_y),
397        );
398
399        bubbles_upper_y += 70;
400
401        let mut bubbles_mini_chart = ChartBuilder::on(&bubbles_temp_root)
402            .x_label_area_size(params.area_margin.x)
403            .y_label_area_size(params.area_margin.y)
404            .build_cartesian_2d(
405                x_min..x_max,
406                0..max_data_len, /*0..30*/ /*mini_chart_axis_y_range.clone()*/
407            )
408            .unwrap();
409
410        bubbles_mini_chart
411            .configure_mesh()
412            .disable_y_mesh()
413            .disable_y_axis()
414            .axis_style(params.colors.axis)
415            .draw()
416            .unwrap();
417
418        // Integer conversion will floor, so when hardly any DATA is transferred
419        // avoid 0, which would cause a divide-by-0 later.
420        let bucket_size = if max_data_len >= 5 {
421            max_data_len / 5
422        } else {
423            1
424        };
425
426        let mut bubbles: Vec<Circle<(f32, i32), i32>> = vec![];
427        let mut lines: Vec<PathElement<(f32, i32)>> = vec![];
428        let mut lines2: Vec<PathElement<(f32, u64)>> = vec![];
429
430        let bubble_sizes = [1, 6, 11, 16, 21, 26, 31];
431
432        for (time, length) in data_frames {
433            // TODO: replace with div_ceil once stablized- https://github.com/rust-lang/rust/issues/88581
434            let i: usize = (length / bucket_size) as usize +
435                usize::from(length % bucket_size != 0);
436            let i = i.saturating_sub(1);
437
438            // TODO: there seems to be a bug if the bubble is greater th x_max it
439            // would still apear. So let's just avoid that.
440            if time > &x_max {
441                continue;
442            }
443
444            lines.push(PathElement::new(
445                [(*time, 2), (*time, 2 + bubble_sizes[i])],
446                style,
447            ));
448
449            lines2.push(PathElement::new([(*time, 0), (*time, *length)], style));
450
451            bubbles.push(Circle::new((*time, 2), bubble_sizes[i], style));
452        }
453
454        // bubbles_mini_chart.draw_series(bubbles).unwrap();
455        bubbles_mini_chart.draw_series(lines2).unwrap();
456    }
