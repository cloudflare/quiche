27//! Pending charts
28//! A single image file containing a view of pending vs. in-flight requests
29
30use plotters::prelude::*;
31
32use crate::datastore::{
33    self,
34};
35
36use crate::plots::*;
37
38use crate::datastore::Datastore;
39use crate::seriesstore::SeriesStore;
40
41#[derive(Debug, Default)]
42pub struct PendingPlotParams {
43    pub clamp: ClampParams,
44    pub chart_size: ChartSize,
45    pub colors: PlotColors,
46    pub display_chart_title: bool,
47}
48
49struct PendingStack {
50    pub time: f32,
51    pub pending: i32,
52    pub in_flight: i32,
53}
54
55pub fn plot_pending(
56    params: &PendingPlotParams, filename: &str, ss: &SeriesStore, ds: &Datastore,
57    ty: &ChartOutputType,
58) {
59    // TODO: put some stuff in series store
60    let x_max = match ds.vantage_point {
61        datastore::VantagePoint::Client => ss.received_x_max,
62        datastore::VantagePoint::Server => ss.sent_x_max,
63    };
64
65    let mut y_max = 0;
66
67    let chart_config = ChartConfig {
68        title: "pending".into(),
69        input_filename: filename.into(),
70        clamp: params.clamp.clone(),
71        app_proto: ds.application_proto,
72        host: ds.host.clone(),
73        session_id: ds.session_id,
74        ty: ty.clone(),
75    };
76
77    chart_config.init_chart_dir();
78
79    let mut x = 0f32;
80    let step_size = 1f32;
81
82    let mut series = vec![];
83
84    while x < x_max {
85        let mut pending = 0;
86        let mut in_flight = 0;
87
88        for req in ds.http_requests.values() {
89            match (
90                req.time_discovery,
91                req.time_first_headers_tx,
92                req.time_fin_rx,
93            ) {
94                (Some(dt), Some(hdrs_tx), Some(fin)) => {
95                    if x >= dt && x < hdrs_tx && x < fin {
96                        pending += 1;
97                    } else if x >= hdrs_tx && x < fin {
98                        in_flight += 1;
99                    }
100                },
101
102                (Some(dt), None, None) =>
103                    if x >= dt {
104                        pending += 1;
105                    },
106
107                (Some(dt), Some(hdrs_tx), None) =>
108                    if x >= dt {
109                        pending += 1;
110                    } else if x >= hdrs_tx {
111                        in_flight += 1;
112                    },
113
114                _ => (),
115            }
116        }
117        series.push(PendingStack {
118            time: x,
119            pending,
120            in_flight,
121        });
122        x += step_size;
123
124        y_max = y_max.max(pending + in_flight);
125    }
126
127    // let size = ChartSize{ width: 1600, height: 600};
128
129    #[cfg(not(target_arch = "wasm32"))]
130    let chart_path = chart_config.chart_filepath();
131
132    #[cfg(not(target_arch = "wasm32"))]
133    let root = make_chart_bitmap_area(
134        &chart_path,
135        params.chart_size,
136        params.colors,
137        ChartMargin::default(),
138    );
139
140    #[cfg(target_arch = "wasm32")]
141    let canvas_id: String = chart_config.canvas_id().unwrap_or_default();
142
143    #[cfg(target_arch = "wasm32")]
144    let root =
145        make_chart_canvas_area(&canvas_id, params.colors, ChartMargin::default());
146
147    let mut builder = ChartBuilder::on(&root);
148    builder
149        .set_label_area_size(LabelAreaPosition::Left, 40)
150        .set_label_area_size(LabelAreaPosition::Bottom, 40);
151
152    if params.display_chart_title {
153        let caption =
154            format!("{} Pending vs. In-flight requests (stacked)", filename);
155        builder.caption(caption, chart_title_style(&params.colors.caption));
156    }
157
158    let mut ctx = builder.build_cartesian_2d(0f32..x_max, 0..y_max).unwrap();
159
160    ctx.configure_mesh()
161        .axis_style(params.colors.axis)
162        .bold_line_style(params.colors.bold_line.mix(0.5))
163        .light_line_style(params.colors.light_line.mix(0.2))
164        .label_style(chart_label_style(&params.colors.caption))
165        .draw()
166        .unwrap();
167
168    ctx.draw_series(series.iter().map(|stack| {
169        // make sure bars are narrow enough that they don't bleed over each
170        // step. If we don't we'll miss periods where there's no requests in
171        // flight.
172        let shrink = step_size / 4f32;
173        let x0 = stack.time + shrink;
174        let x1 = x0 + (step_size - shrink);
175
176        // Clippy wants to remove the variable assignment that helps understand
177        // WTF this stuff is doing, screw clippy.
178        #[allow(clippy::let_and_return)]
179        let in_flight_bar = Rectangle::new(
180            [(x0, 0), (x1, stack.in_flight)],
181            colors::FOREST_GREEN.filled(),
182        );
183        in_flight_bar
184    }))
185    .unwrap()
186    .label("In-flight requests")
187    .legend(|(x, y)| {
188        PathElement::new(vec![(x, y), (x + 20, y)], colors::FOREST_GREEN)
189    });
190
191    ctx.draw_series(series.iter().map(|stack| {
192        // make sure bars are narrow enough that they don't bleed over each
193        // step. If we don't we'll miss periods where there's no requests in
194        // flight.
195        let shrink = step_size / 3f32;
196        let x0 = stack.time + shrink;
197        let x1 = x0 + (step_size - shrink);
198
199        // pending is stack on top of in-flight
200        // Clippy wants to remove the variable assignment that helps understand
201        // WTF this stuff is doing, screw clippy.
202        #[allow(clippy::let_and_return)]
203        let pending_bar = Rectangle::new(
204            [(x0, stack.in_flight), (x1, stack.in_flight + stack.pending)],
205            colors::ORANGE.filled(),
206        );
207        pending_bar
208    }))
209    .unwrap()
210    .label("Pending requests")
211    .legend(|(x, y)| PathElement::new(vec![(x, y), (x + 20, y)], colors::ORANGE));
212
213    ctx.configure_series_labels()
214        .label_font(chart_label_style(&params.colors.caption))
215        .background_style(params.colors.fill.mix(0.8))
216        .border_style(params.colors.axis)
217        .position(SeriesLabelPosition::UpperRight)
218        .draw()
219        .unwrap();
220}
