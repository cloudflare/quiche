40const Y_WIGGLE: f64 = 1.2;
41
42struct XYMinMax {
43    pub x: XMinMax,
44    pub y_min: u64,
45    pub y_max: u64,
46}
47
48impl XYMinMax {
49    fn init(params: &PlotParameters, ss: &SeriesStore, ds: &Datastore) -> Self {
50        let x = XMinMax::new(
51            ss.received_x_min,
52            ss.received_x_max,
53            params.clamp.start,
54            params.clamp.end,
55        );
56
57        Self {
58            x,
59            y_min: 0,
60            y_max: Self::y_max(ss, &ds.application_proto),
61        }
62    }
63
64    fn y_range(&self) -> std::ops::Range<u64> {
65        self.y_min..self.y_max
66    }
67
68    fn y_max(ss: &SeriesStore, proto: &ApplicationProto) -> u64 {
69        let y = match proto {
70            ApplicationProto::Http2 =>
71                *(ss.h2_send_window_absolute_max.get(&0).unwrap_or(&0)),
72            ApplicationProto::Http3 =>
73                ss.netlog_quic_stream_received_connection_cumulative
74                    .last()
75                    .unwrap_or(&(0.0, 0))
76                    .1,
77        };
78
79        (y as f64 * Y_WIGGLE) as u64
80    }
81}

83fn blocked_lines<DB: DrawingBackend>(
84    ds: &Datastore, y_max: u64,
85    chart: &mut ChartContext<'_, DB, Cartesian2d<RangedCoordf32, RangedCoordu64>>,
86) {
87    if let Some(blocked) = ds.netlog_quic_server_window_blocked.get(&-1).cloned()
88    {
89        let blocked_lines = blocked
90            .iter()
91            .map(|t| PathElement::new([(*t, 0), (*t, y_max)], GREEN));
92
93        chart
94            .draw_series(blocked_lines)
95            .unwrap()
96            .label("server conn blocked")
97            .legend(|(x, y)| PathElement::new(vec![(x, y), (x + 20, y)], GREEN));
98    }
99}

101fn received_data<DB: DrawingBackend>(
102    ss: &SeriesStore, proto: &ApplicationProto,
103    chart: &mut ChartContext<'_, DB, Cartesian2d<RangedCoordf32, RangedCoordu64>>,
104) {
105    let points = match proto {
106        ApplicationProto::Http2 =>
107            ss.netlog_h2_stream_received_connection_cumulative.clone(),
108        ApplicationProto::Http3 =>
109            ss.netlog_quic_stream_received_connection_cumulative.clone(),
110    };
111
112    chart
113        .draw_series(LineSeries::new(points, ORANGE))
114        .unwrap()
115        .label("cumulative received stream data")
116        .legend(|(x, y)| PathElement::new(vec![(x, y), (x + 20, y)], ORANGE));
117}

119fn window_updates<DB: DrawingBackend>(
120    ss: &SeriesStore, ds: &Datastore,
121    chart: &mut ChartContext<'_, DB, Cartesian2d<RangedCoordf32, RangedCoordu64>>,
122) {
123    let points = match ds.application_proto {
124        ApplicationProto::Http2 => {
125            if let Some(conn_win_updates) =
126                ss.h2_send_window_series_absolute.get(&0).cloned()
127            {
128                conn_win_updates
129            } else {
130                return;
131            }
132        },
133
134        ApplicationProto::Http3 => {
135            if let Some(conn_win_updates) =
136                ss.netlog_quic_client_side_window_updates.get(&-1).cloned()
137            {
138                conn_win_updates
139            } else {
140                return;
141            }
142        },
143    };
144
145    // Help us see the exact time a window update happened.
146    let circles: Vec<Circle<(f32, u64), i32>> = points
147        .iter()
148        .map(|point| Circle::new(*point, 2, BLUE))
149        .collect();
150
151    chart
152        .draw_series(LineSeries::new(points, BLUE))
153        .unwrap()
154        .label("sent window updates")
155        .legend(|(x, y)| PathElement::new(vec![(x, y), (x + 20, y)], BLUE));
156
157    chart.draw_series(circles).unwrap();
158}

160fn draw_series<'a, DB: DrawingBackend + 'a>(
161    chart: &mut ChartContext<'a, DB, Cartesian2d<RangedCoordf32, RangedCoordu64>>,
162    params: &PlotParameters, ss: &SeriesStore, ds: &Datastore, axis: XYMinMax,
163) {
164    draw_mesh(
165        &params.colors,
166        "Relative time (ms)",
167        "Bytes",
168        params.display_minor_lines,
169        chart,
170    );
171
172    // Draw the series
173    blocked_lines(ds, axis.y_max, chart);
174    received_data(ss, &ds.application_proto, chart);
175    window_updates(ss, ds, chart);
176
177    if params.display_legend {
178        // Draw the series legend
179        chart
180            .configure_series_labels()
181            .label_font(chart_label_style(&params.colors.caption))
182            .background_style(params.colors.fill.mix(0.8))
183            .border_style(params.colors.axis)
184            .position(SeriesLabelPosition::LowerRight)
185            .draw()
186            .unwrap();
187    }
188}

190#[cfg(target_arch = "wasm32")]
191pub fn plot_conn_flow_control_canvas<'a>(
192    params: &PlotParameters, filename: &str, ss: &SeriesStore, ds: &Datastore,
193    ty: &ChartOutputType,
194) -> ChartContext<'a, CanvasBackend, Cartesian2d<RangedCoordf32, RangedCoordu64>>
195{
196    let chart_config =
197        make_chart_config("flow_control", params, filename, ds, ty);
198
199    let canvas_id: String = chart_config.canvas_id().unwrap_or_default();
200    let root =
201        make_chart_canvas_area(&canvas_id, params.colors, params.chart_margin);
202
203    let axis = XYMinMax::init(params, ss, ds);
204
205    let mut builder = ChartBuilder::on(&root);
206    builder
207        .x_label_area_size(params.area_margin.x)
208        .y_label_area_size(params.area_margin.y);
209
210    if params.display_chart_title {
211        let caption = format!("{} Connection Flow Control timeline", filename);
212        builder.caption(caption, chart_title_style(&params.colors.caption));
213    }
214
215    let mut fc_chart = builder
216        .build_cartesian_2d(axis.x.range(), axis.y_range())
217        .unwrap();
218
219    draw_series(&mut fc_chart, params, ss, ds, axis);
220
221    fc_chart
222}

224#[cfg(not(target_arch = "wasm32"))]
225pub fn plot_conn_flow_control(
226    params: &PlotParameters, filename: &str, ss: &SeriesStore, ds: &Datastore,
227    ty: &ChartOutputType,
228) {
229    let chart_config =
230        make_chart_config("flow_control", params, filename, ds, ty);
231    let chart_path = chart_config.chart_filepath();
232
233    let root = make_chart_bitmap_area(
234        &chart_path,
235        params.chart_size,
236        params.colors,
237        params.chart_margin,
238    );
239
240    let axis = XYMinMax::init(params, ss, ds);
241
242    let mut builder = ChartBuilder::on(&root);
243    builder
244        .x_label_area_size(params.area_margin.x)
245        .y_label_area_size(params.area_margin.y);
246
247    if params.display_chart_title {
248        let caption = format!("{} Connection Flow Control timeline", filename);
249        builder.caption(caption, chart_title_style(&params.colors.caption));
250    }
251
252    let mut fc_chart = builder
253        .build_cartesian_2d(axis.x.range(), axis.y_range())
254        .unwrap();
255
256    draw_mesh(
257        &params.colors,
258        "Relative time (ms)",
259        "Bytes",
260        params.display_minor_lines,
261        &mut fc_chart,
262    );
263
264    draw_series(&mut fc_chart, params, ss, ds, axis);
