266    fn packet_sent(&mut self, data_store: &Datastore) {
267        if let Some(onertt_pkts) =
268            &data_store.packet_sent.get(&crate::PacketType::OneRtt)
269        {
270            // TODO: perhaps better to take these counts when processing recovery
271            // metrics
272            let mut sent_count = 0;
273            let mut delivered_count = 0;
274            let mut lost_count = 0;
275
276            for (pkt_num, pkt_info) in onertt_pkts.iter() {
277                sent_count += 1;
278                push_interp(
279                    &mut self.onertt_packet_created,
280                    (pkt_info.created_time, *pkt_num),
281                );
282                push_interp(
283                    &mut self.onertt_packet_sent_aggregate_count,
284                    (pkt_info.created_time, sent_count),
285                );
286
287                // Hacky way to detect lost packets. We don't have the actual
288                // time the loss happened, so just reuse the packet creation time
289                if pkt_info.acked.is_none() {
290                    self.onertt_packet_lost_hacky
291                        .push((pkt_info.created_time, *pkt_num));
292                    lost_count += 1;
293                    push_interp(
294                        &mut self.onertt_packet_lost_aggregate_count,
295                        (pkt_info.created_time, lost_count),
296                    );
297                } else {
298                    delivered_count += 1;
299                    push_interp(
300                        &mut self.onertt_packet_delivered_aggregate_count,
301                        (pkt_info.created_time, delivered_count),
302                    );
303                }
304
305                self.y_max_onertt_pkt_sent_plot =
306                    std::cmp::max(self.y_max_onertt_pkt_sent_plot, *pkt_num);
307
308                // send_at_time is optional
309                if let Some(packet_sent_at) = pkt_info.send_at_time {
310                    push_interp(
311                        &mut self.onertt_packet_sent,
312                        (packet_sent_at, *pkt_num),
313                    );
314
315                    let delta = packet_sent_at - pkt_info.created_time;
316                    push_interp(
317                        &mut self.onertt_packet_created_sent_delta,
318                        (*pkt_num, delta),
319                    );
320
321                    // std::cmp hates floats, so go old school
322                    if delta > self.y_max_onertt_packet_created_sent_delta {
323                        self.y_max_onertt_packet_created_sent_delta = delta;
324                    }
325
326                    if delta < self.y_min_onertt_packet_created_sent_delta {
327                        self.y_min_onertt_packet_created_sent_delta = delta;
328                    }
329                }
330            }
331        }
332    }

349    fn missing_packets(&mut self, data_store: &Datastore) {
350        // Simple way to try and determine if new missing packets were logged.
351        // TODO: netlog produces an array of missing_packets by packet number.
352        // The array can grow or shrink as QUIC connection evolves. We should
353        // only insert an event when a new unique value is observed.
354        // TODO: rewrite to avoid copies
355        let mut last: Vec<u64> = vec![];
356
357        for (event_time, missing_pkts) in
358            &data_store.netlog_ack_sent_missing_packets_raw
359        {
360            if &last != missing_pkts {
361                last = missing_pkts.clone();
362
363                self.netlog_missing_packets.push(*event_time);
364            }
365        }
366    }

526    fn received_data_frames(&mut self, data_store: &Datastore) {
527        for (stream, points) in &data_store.received_data_frames {
528            let mut series_points = vec![];
529
530            // insert a dummy 0'th point at the first incidence time
531            if let Some(first) = points.first() {
532                series_points.push((first.0, 0));
533            } else {
534                continue;
535            }
536
537            let mut last_y = series_points.first().unwrap().1;
538
539            for point in points {
540                let new_y = last_y + point.1;
541
542                self.received_x_max = self.received_x_max.max(point.0);
543                self.update_stream_recv_y_axis_max(new_y);
544
545                series_points.push((point.0, new_y));
546
547                last_y = new_y;
548            }
549
550            self.received_data_max
551                .insert(*stream, series_points.last().unwrap().1);
552
553            self.received_data_frames_series
554                .insert(*stream, series_points);
555        }
556    }

565            } else {
566                continue;
567            }

569            let mut last_y = series_points.first().unwrap().1;

571            for point in points {
572                let new_y = last_y + point.1;
573
574                self.update_sent_x_axis_max(point.0);

577                series_points.push((point.0, new_y));
578
579                last_y = new_y;
580            }
581
582            self.sent_data_max
583                .insert(*stream, series_points.last().unwrap().1);
584
585            self.sent_data_frames_series.insert(*stream, series_points);
586        }

589    // TODO: hack. Initial window size can theoretically vary through
590    // the connection, which makes calculations hard. Practically,
591    // nobody does that, so just assume the initial window was either
592    // indicated in settings or omitted in order to use defaults.
593    fn initial_h2_fc_value(stream_id: u32, data_store: &Datastore) -> u32 {
594        if stream_id == 0 {
595            // https://www.rfc-editor.org/rfc/rfc9113.html#section-6.9.2
596            // Connection-level flow control is 65,535 octets
597            H2_DEFAULT_WINDOW_SIZE
598        } else {
599            data_store
600                .h2_server_settings
601                .initial_window_size
602                .unwrap_or(H2_DEFAULT_WINDOW_SIZE)
603        }
604    }

607    fn h2_fc_balanced(&mut self, data_store: &Datastore) {
608        for (stream, points) in &data_store.h2_send_window_updates_balanced {
609            let mut series_points = vec![];
610            let mut y_max = 0;
611
612            let initial_window =
613                Self::initial_h2_fc_value(*stream, data_store) as i32;
614
615            // insert a dummy 0'th point at the first incidence time
616            if let Some(first) = points.first() {
617                series_points.push((first.0, initial_window));
618            } else {
619                continue;
620            }
621
622            let mut last_y = series_points.first().unwrap().1;
623
624            for point in points {
625                let new_y = last_y + point.1;
626                self.update_sent_x_axis_max(point.0);
627                y_max = y_max.max(new_y);
628
629                push_interp(&mut series_points, (point.0, new_y));
630                last_y = new_y;
631            }
632
633            self.h2_send_window_series_balanced
634                .insert(*stream, series_points);
635            self.h2_send_window_balanced_max.insert(*stream, y_max);
636        }
637    }

652            }

663                last_y = new_y;
664            }
