37macro_rules! printy {
38    ($k:expr, $value:expr, $s:expr) => {{
39        $s += &format!("{}={}, ", $k, $value);
40    }};
41}
42
43macro_rules! printyo {
44    ($k:expr, $value:expr, $s:expr) => {{
45        if let Some(v) = $value {
46            $s += &format!("{}={}, ", $k, v);
47        }
48    }};
49}
50
51macro_rules! printy_json {
52    ($k:expr, $value:expr, $s:expr) => {{
53        $s += &format!(
54            "{}={}, ",
55            $k,
56            &serde_json::to_string(&$value).unwrap().replace("\"", "")
57        );
58    }};
59}
60
61macro_rules! printyo_json {
62    ($k:expr, $value:expr, $s:expr) => {{

64            $s += &format!(
65                "{}={}, ",
66                $k,
67                &serde_json::to_string(&v).unwrap().replace("\"", "")
68            );
69        }
70    }};
71}

82pub fn frames_to_string(frames: &[QuicFrame]) -> String {
83    let mut s = String::new();
84    for f in frames {
85        match f {
86            qlog::events::quic::QuicFrame::Padding { payload_length, .. } => {
87                s += &format!(" PADDING {{len={payload_length}}}");
88            },
89            qlog::events::quic::QuicFrame::Ping { .. } => {
90                s += " PING";
91            },
92            qlog::events::quic::QuicFrame::Ack { acked_ranges, .. } => {
93                s += " ACK {";
94                if let Some(ar) = acked_ranges {
95                    match ar {
96                        AckedRanges::Single(items) => {
97                            for a in items {
98                                for b in a {
99                                    s += &format!{"{b}, "};
100                                }
101                            }
102                        },
103                        AckedRanges::Double(items) => {
104                            for a in items {
105                                s += &format!{"{}-{}, ", a.0, a.1};
106                            }
107                        },
108                    }
109                }
110                s += "}";
111            },
112            qlog::events::quic::QuicFrame::ResetStream { stream_id, error_code, final_size, .. } => {
113                s += &format!(" RESET_STREAM {{id={stream_id}, error_code={error_code}, final_size={final_size}}}");
114            },
115            qlog::events::quic::QuicFrame::StopSending { stream_id, error_code, ..} => {
116                s += &format!(" STOP_SENDING {{id={stream_id}, error_code={error_code}}}");
117            },
118            qlog::events::quic::QuicFrame::Crypto { offset, length } => {
119                s += &format!(" CRYPTO {{off={offset}, len={length}}}");
120            },
121            qlog::events::quic::QuicFrame::NewToken { token } => {
122               s += " NEW_TOKEN ";
123               if let Some(ty) = &token.ty {
124                    s += &format!("{{ty={ty:?}}}");
125               }
126            },
127            qlog::events::quic::QuicFrame::Stream { stream_id, offset, length, fin, .. } => {
128                s += &format!(" STREAM {{id={stream_id}, off={offset}, len={length}");
129                if let Some(f) = fin {
130                    s += &format!(", fin={f}")
131                }
132                s += "}";
133            },
134            qlog::events::quic::QuicFrame::MaxData { maximum } => {
135                s += &format!(" MAX_DATA {{max={maximum}}}");
136            },
137            qlog::events::quic::QuicFrame::MaxStreamData { stream_id, maximum } => {
138                s += &format!(" MAX_STREAM_DATA {{id={stream_id}, max={maximum}}}");
139            },
140            qlog::events::quic::QuicFrame::MaxStreams { stream_type, maximum } => {
141                s += &format!(" MAX_STREAMS {{ty={stream_type:?}, max={maximum}}}");
142            },
143            qlog::events::quic::QuicFrame::DataBlocked { limit } => {
144                s += &format!(" DATA_BLOCKED {{limit={limit}}}");
145            },
146            qlog::events::quic::QuicFrame::StreamDataBlocked { stream_id, limit } => {
147                s += &format!(" STREAM_DATA_BLOCKED {{id={stream_id}, limit={limit}}}");
148            },
149            qlog::events::quic::QuicFrame::StreamsBlocked { stream_type, limit } => {
150                s += &format!(" STREAMS_BLOCKED {{ty={stream_type:?}, limit={limit}}}");
151            },
152            qlog::events::quic::QuicFrame::NewConnectionId { /*sequence_number, retire_prior_to, connection_id_length, connection_id, stateless_reset_token*/ .. } => {
153                s += " NEW_CONNECTION_ID {{todo='todo'}}";
154            },
155            qlog::events::quic::QuicFrame::RetireConnectionId { sequence_number } => {
156                s += &format!(" RETIRE_CONNECION_ID {{sn={sequence_number}}}");
157            },
158            qlog::events::quic::QuicFrame::PathChallenge { /*data*/ .. } => {
159                s += " PATH_CHALLENGE {{todo='todo'}}";
160            },
161            qlog::events::quic::QuicFrame::PathResponse { /*data*/ .. } => {
162                s += " PATH_RESPONSE {{todo='todo'}}";
163            },
164            qlog::events::quic::QuicFrame::ConnectionClose { error_space, error_code, reason, .. } => {
165               s += " CONNECTION_CLOSE {";
166               if let Some(es) = error_space {
167                    s += &format!(" ty={es:?},");
168               }
169               printyo!(" code", error_code, s);
170               printyo!(" reason", reason, s);
171               s += "}";
172            },
173            qlog::events::quic::QuicFrame::HandshakeDone => {
174                s += " HANDSHAKE_DONE";
175            },
176            qlog::events::quic::QuicFrame::Datagram { length, .. } => {
177               s += &format!(" DATAGRAM {{len={length}}}");
178            },
179            qlog::events::quic::QuicFrame::Unknown { raw_frame_type, .. } => {
180               s += &format!(" UNKNOWN {{raw_type={raw_frame_type}}}");
181            },
182        }
183    }
184
185    s
186}

188fn http_frame_to_string(frame: &Http3Frame) -> String {
189    let mut s = String::new();
190
191    match frame {
192        Http3Frame::Data { raw } =>{
193            s += " DATA";
194            if let Some(r) = raw {
195                printyo!("len", r.length, s);
196            }
197        },
198        Http3Frame::Headers { headers } => {
199            s += " HEADERS {";
200
201            for header in headers {
202                s += &format!("{}: {}, ", header.name, header.value);
203            }
204
205            s += "}";
206        },
207        Http3Frame::CancelPush { push_id } => {
208            s += &format!(" CANCEL_PUSH {{id={push_id}}}");
209        },
210        Http3Frame::Settings { /* settings */ ..} => {
211            s += " SETTINGS {{todo}}";
212        }
213        Http3Frame::PushPromise { /*push_id, headers */ ..} => {
214            s += " PUSH_PROMISE {{todo}}";
215        }
216        Http3Frame::Goaway { id } => {
217            s += &format!(" GOAWAY {{id={id}}}");
218        }
219        Http3Frame::MaxPushId { push_id } => {
220            s += &format!(" MAX_PUSH_ID {{id={push_id}}}");
221        }
222        Http3Frame::PriorityUpdate { /*target_stream_type, prioritized_element_id, priority_field_value*/ .. } => {
223            s += " PRIORITY_UPDATE {{todo}}";
224        },
225        Http3Frame::Reserved { /*length*/ .. } => {
226            s += " GREASE {{todo}}";
227        },
228        Http3Frame::Unknown { frame_type_value, .. } => {
229            s += &format!(" UNKNOWN {{ty={frame_type_value}}}");
230        }
231    }
232
233    s
234}

236pub fn sqlog_event_list(
237    events: &[qlog::reader::Event],
238) -> tabled::builder::Builder {
239    let mut pp = vec![];
240
241    for event in events {
242        match event {
243            qlog::reader::Event::Qlog(ev) => {
244                let (cat, ty) = category_and_type_from_event(ev);
245
246                let mut p = PrintableEvent {
247                    time: ev.time,
248                    ty,
249                    category: cat,
250                    ..Default::default()
251                };
252
253                match &ev.data {
254                    EventData::ConnectionStarted(v) => {
255                        printyo!("protocol", &v.protocol, p.details);
256                        printyo!("ip_version", &v.ip_version, p.details);
257                        printy!("src_ip", v.src_ip, p.details);
258                        printyo!("src_port", v.src_port, p.details);
259                        printyo!("src_cid", &v.src_cid, p.details);
260                        printy!("dst_ip", v.dst_ip, p.details);
261                        printyo!("dst_port", v.dst_port, p.details);
262                        printyo!("dst_cid", &v.dst_cid, p.details);
263                    },
264                    EventData::ConnectionClosed(v) => {
265                        printyo_json!("owner", &v.owner, p.details);
266                        printyo_json!(
267                            "connection_code",
268                            &v.connection_code,
269                            p.details
270                        );
271                        printyo_json!(
272                            "application_code",
273                            &v.application_code,
274                            p.details
275                        );
276                        printyo!("internal_code", &v.internal_code, p.details);
277                        printyo!("reason", &v.reason, p.details);
278                        printyo_json!("trigger", &v.trigger, p.details);
279                    },
280                    EventData::TransportParametersSet(v) => {
281                        printyo_json!("owner", &v.owner, p.details);
282                        printyo!(
283                            "resumption_allowed",
284                            &v.resumption_allowed,
285                            p.details
286                        );
287                        printyo!(
288                            "early_data_enabled",
289                            &v.early_data_enabled,
290                            p.details
291                        );
292                        printyo!("tls_cipher", &v.tls_cipher, p.details);
293                        printyo!(
294                            "aead_tag_length",
295                            &v.aead_tag_length,
296                            p.details
297                        );
298                        printyo!(
299                            "odcid",
300                            &v.original_destination_connection_id,
301                            p.details
302                        );
303                        printyo!(
304                            "initial_scid",
305                            &v.initial_source_connection_id,
306                            p.details
307                        );
308                        printyo!(
309                            "retry_scid",
310                            &v.retry_source_connection_id,
311                            p.details
312                        );
313                        printyo!(
314                            "stateless_reset_token",
315                            &v.stateless_reset_token,
316                            p.details
317                        );
318                        printyo!(
319                            "disable_active_migration",
320                            &v.disable_active_migration,
321                            p.details
322                        );
323                        printyo!(
324                            "max_idle_timeout",
325                            &v.max_idle_timeout,
326                            p.details
327                        );
328                        printyo!(
329                            "max_udp_payload_size",
330                            &v.max_udp_payload_size,
331                            p.details
332                        );
333                        printyo!("max_ack_delay", &v.max_ack_delay, p.details);
334                        printyo!(
335                            "active_connection_id_limit",
336                            &v.active_connection_id_limit,
337                            p.details
338                        );
339                        printyo!(
340                            "initial_max_data",
341                            &v.initial_max_data,
342                            p.details
343                        );
344                        printyo!(
345                            "initial_max_stream_data_bidi_local",
346                            &v.initial_max_stream_data_bidi_local,
347                            p.details
348                        );
349                        printyo!(
350                            "initial_max_stream_data_bidi_remote",
351                            &v.initial_max_stream_data_bidi_remote,
352                            p.details
353                        );
354                        printyo!(
355                            "initial_max_stream_data_uni",
356                            &v.initial_max_stream_data_uni,
357                            p.details
358                        );
359                        printyo!(
360                            "initial_max_streams_bidi",
361                            &v.initial_max_streams_bidi,
362                            p.details
363                        );
364                        printyo!(
365                            "initial_max_streams_uni",
366                            &v.initial_max_streams_uni,
367                            p.details
368                        );
369                        printyo_json!(
370                            "preferred_address",
371                            &v.preferred_address,
372                            p.details
373                        );
374                        // TODO: v.unknown_parameters
375                    },
376                    EventData::PacketSent(v) => {
377                        p.details +=
378                            &serde_json::to_string(&v.header.packet_type)
379                                .unwrap()
380                                .replace("\"", "");
381                        printyo!(" pn", v.header.packet_number, p.details);
382
383                        if let Some(frames) = &v.frames {
384                            p.details += &frames_to_string(frames);
385                        }
386                    },
387                    EventData::PacketReceived(v) => {
388                        p.details +=
389                            &serde_json::to_string(&v.header.packet_type)
390                                .unwrap()
391                                .replace("\"", "");
392                        printyo!(" pn", v.header.packet_number, p.details);
393
394                        if let Some(frames) = &v.frames {
395                            p.details += &frames_to_string(frames);
396                        }
397                    },
398                    EventData::DataMoved(v) => {
399                        printyo!("id", v.stream_id, p.details);
400                        printyo!("off", v.offset, p.details);
401                        printyo!("len", v.length, p.details);
402                        printyo_json!("from", &v.from, p.details);
403                        printyo_json!("to", &v.to, p.details);
404                    },
405                    EventData::MetricsUpdated(v) => {
406                        printyo!("min_rtt", v.min_rtt, p.details);
407                        printyo!("smoothed_rtt", v.smoothed_rtt, p.details);
408                        printyo!("latest_rtt", v.latest_rtt, p.details);
409                        printyo!("rtt_variance", v.rtt_variance, p.details);
410                        printyo!("pto_count", v.pto_count, p.details);
411                        printyo!("cwnd", v.congestion_window, p.details);
412                        printyo!("bytes_in_flight", v.bytes_in_flight, p.details);
413                        printyo!("ssthresh", v.ssthresh, p.details);
414                        printyo!(
415                            "packets_in_flight",
416                            v.packets_in_flight,
417                            p.details
418                        );
419                        printyo!("pacing_rate", v.pacing_rate, p.details);
420                    },
421                    EventData::H3StreamTypeSet(ev) => {
422                        printy!("id", &ev.stream_id, p.details);
423                        printyo_json!("owner", &ev.owner, p.details);
424                        printy_json!("ty", &ev.stream_type, p.details);
425                    },
426                    EventData::H3FrameCreated(v) => {
427                        printy!("id", v.stream_id, p.details);
428                        printyo!("len", v.length, p.details);
429                        p.details += &http_frame_to_string(&v.frame);
430                    },
431                    EventData::H3FrameParsed(v) => {
432                        printy!("id", v.stream_id, p.details);
433                        printyo!("len", v.length, p.details);
434                        p.details += &http_frame_to_string(&v.frame);
435                    },
436
437                    _ => {
438                        p.details += &serde_json::to_string(&ev.data).unwrap();
439                    },
440                }
441
442                pp.push(p);
443            },
444
445            qlog::reader::Event::Json(ev) => {
446                let (cat, ty) = category_and_type_from_name(&ev.name);
447
448                let p = PrintableEvent {
449                    time: ev.time,
450                    ty,
451                    category: cat,
452                    details: serde_json::to_string(&ev.data).unwrap(),
453                };
454
455                pp.push(p);
456            },
457        }
458    }
459
460    Table::builder(pp)
