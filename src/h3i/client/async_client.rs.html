77/// Connect to the socket.
78pub async fn connect(
79    args: &H3iConfig, frame_actions: Vec<Action>,
80    close_trigger_frames: Option<CloseTriggerFrames>,
81) -> std::result::Result<BuildingConnectionSummary, ClientError> {
82    let quic_settings = create_config(args);
83    let mut connection_params =
84        ConnectionParams::new_client(quic_settings, None, Hooks::default());
85
86    connection_params.session = args.session.clone();
87
88    let ParsedArgs {
89        connect_url,
90        bind_addr,
91        peer_addr,
92    } = parse_args(args);
93
94    let socket = tokio::net::UdpSocket::bind(bind_addr).await.unwrap();
95    socket.connect(peer_addr).await.unwrap();
96
97    log::info!(
98        "connecting to {:} from {:}",
99        peer_addr,
100        socket.local_addr().unwrap()
101    );
102
103    let (h3i, conn_summary_fut) =
104        H3iDriver::new(frame_actions, close_trigger_frames);
105    match tokio_quiche::quic::connect_with_config(
106        Socket::try_from(socket).unwrap(),
107        connect_url,
108        &connection_params,
109        h3i,
110    )
111    .await
112    {
113        Ok(_) => Ok(conn_summary_fut),
114        Err(_) => Err(ClientError::HandshakeFail),
115    }
116}

118fn create_config(args: &H3iConfig) -> QuicSettings {
119    let mut quic_settings = QuicSettings::default();
120
121    quic_settings.verify_peer = args.verify_peer;
122    quic_settings.max_idle_timeout =
123        Some(Duration::from_millis(args.idle_timeout));
124    quic_settings.max_recv_udp_payload_size = MAX_DATAGRAM_SIZE;
125    quic_settings.max_send_udp_payload_size = MAX_DATAGRAM_SIZE;
126    quic_settings.initial_max_data = 10_000_000;
127    quic_settings.initial_max_stream_data_bidi_local =
128        args.max_stream_data_bidi_local;
129    quic_settings.initial_max_stream_data_bidi_remote =
130        args.max_stream_data_bidi_remote;
131    quic_settings.initial_max_stream_data_uni = args.max_stream_data_uni;
132    quic_settings.initial_max_streams_bidi = args.max_streams_bidi;
133    quic_settings.initial_max_streams_uni = args.max_streams_uni;
134    quic_settings.disable_active_migration = true;
135    quic_settings.active_connection_id_limit = 0;
136    quic_settings.max_connection_window = args.max_window;
137    quic_settings.max_stream_window = args.max_stream_window;
138    quic_settings.grease = false;
139
140    quic_settings.capture_quiche_logs = true;
141    quic_settings.keylog_file = std::env::var_os("SSLKEYLOGFILE")
142        .and_then(|os_str| os_str.into_string().ok());
143
144    quic_settings
145}

147/// The [`Future`] used to build a [`ConnectionSummary`].
148///
149/// At a high level, [`H3iDriver`] will interact with the UDP socket directly,
150/// sending and receiving data as necessary. As new data is received, it will
151/// send [`ConnectionRecord`]s to this struct, which uses these records to
152/// construct the [`ConnectionSummary`].
153#[must_use = "must await to get a ConnectionSummary"]
154pub struct BuildingConnectionSummary {
155    rx: mpsc::UnboundedReceiver<ConnectionRecord>,
156    summary: Option<ConnectionSummary>,
157    seen_all_close_trigger_frames: Option<oneshot::Sender<()>>,
158}
159
160impl BuildingConnectionSummary {
161    fn new(
162        rx: mpsc::UnboundedReceiver<ConnectionRecord>,
163        close_trigger_frames: Option<CloseTriggerFrames>,
164        trigger_frame_tx: oneshot::Sender<()>,
165    ) -> Self {
166        let summary = ConnectionSummary {
167            stream_map: StreamMap::new(close_trigger_frames),
168            ..Default::default()
169        };
170
171        Self {
172            rx,
173            summary: Some(summary),
174            seen_all_close_trigger_frames: Some(trigger_frame_tx),
175        }
176    }
177}
178
179impl Future for BuildingConnectionSummary {
180    type Output = ConnectionSummary;
181
182    fn poll(
183        mut self: Pin<&mut Self>, cx: &mut Context<'_>,
184    ) -> Poll<Self::Output> {
185        while let Poll::Ready(Some(record)) = self.rx.poll_recv(cx) {
186            // Grab all records received from the current event loop iteration and
187            // insert them into the in-progress summary
188            let summary = self.summary.as_mut().expect("summary already taken");
189
190            match record {
191                ConnectionRecord::StreamedFrame { stream_id, frame } => {
192                    let stream_map = &mut summary.stream_map;
193                    stream_map.insert(stream_id, frame);
194
195                    if stream_map.all_close_trigger_frames_seen() {
196                        // Signal the H3iDriver task to close the connection.
197                        if let Some(expected_tx) =
198                            self.seen_all_close_trigger_frames.take()
199                        {
200                            let _ = expected_tx.send(());
201                        }
202                    }
203                },
204                ConnectionRecord::ConnectionStats(s) => summary.stats = Some(s),
205                ConnectionRecord::PathStats(ps) => summary.path_stats = ps,
206                ConnectionRecord::Close(d) => summary.conn_close_details = d,
207            };
208        }
209
210        if self.rx.is_closed() {
211            // The sender drops when the Tokio-Quiche IOW finishes, so the
212            // connection is done and we're safe to yield the summary.
213            let summary = self.summary.take().expect("summary already taken");
214            Poll::Ready(summary)
215        } else {
216            Poll::Pending
217        }
218    }
219}

221pub struct H3iDriver {
222    buffer: Pooled<ConsumeBuffer>,
223    actions: Vec<Action>,
224    actions_executed: usize,
225    next_fire_time: Instant,
226    waiting_for_responses: WaitingFor,
227    record_tx: mpsc::UnboundedSender<ConnectionRecord>,
228    stream_parsers: StreamParserMap,
229    close_trigger_seen_rx: oneshot::Receiver<()>,
230}
231
232impl H3iDriver {
233    fn new(
234        actions: Vec<Action>, close_trigger_frames: Option<CloseTriggerFrames>,
235    ) -> (Self, BuildingConnectionSummary) {
236        let (record_tx, record_rx) = mpsc::unbounded_channel();
237        let (close_trigger_seen_tx, close_trigger_seen_rx) = oneshot::channel();
238        let fut = BuildingConnectionSummary::new(
239            record_rx,
240            close_trigger_frames,
241            close_trigger_seen_tx,
242        );
243
244        (
245            Self {
246                buffer: BufFactory::get_max_buf(),
247                actions,
248                actions_executed: 0,
249                next_fire_time: Instant::now(),
250                waiting_for_responses: WaitingFor::default(),
251                record_tx,
252                stream_parsers: StreamParserMap::default(),
253                close_trigger_seen_rx,
254            },
255            fut,
256        )
257    }
258
259    /// If the next action should fire.
260    fn should_fire(&self) -> bool {
261        Instant::now() >= self.next_fire_time
262    }
263
264    /// Insert all waits into the waiting set.
265    fn register_waits(&mut self) {
266        while self.actions_executed < self.actions.len() {
267            if let Action::Wait { wait_type } =
268                &self.actions[self.actions_executed]
269            {
270                self.actions_executed += 1;
271
272                match wait_type {
273                    WaitType::WaitDuration(duration) => {
274                        self.next_fire_time = Instant::now() + *duration;
275
276                        log::debug!(
277                            "h3i: waiting for responses: {:?}",
278                            self.waiting_for_responses
279                        );
280                    },
281                    WaitType::StreamEvent(event) => {
282                        self.waiting_for_responses.add_wait(event);
283                    },
284                }
285            } else {
286                break;
287            }
288        }
289    }
290}

292impl Client for H3iDriver {
293    fn stream_parsers_mut(&mut self) -> &mut StreamParserMap {
294        &mut self.stream_parsers
295    }
296
297    fn handle_response_frame(
298        &mut self, stream_id: u64, frame: crate::frame::H3iFrame,
299    ) {
300        self.record_tx
301            .send(ConnectionRecord::StreamedFrame { stream_id, frame })
302            .expect("H3iDriver task dropped")
303    }
304}

306impl ApplicationOverQuic for H3iDriver {
307    fn on_conn_established(
308        &mut self, _qconn: &mut QuicheConnection, _handshake_info: &HandshakeInfo,
309    ) -> QuicResult<()> {
310        log::info!("h3i: HTTP/3 connection established");
311        Ok(())
312    }
313
314    fn should_act(&self) -> bool {
315        // Even if the connection wasn't established, we should still send
316        // terminal records to the summary
317        true
318    }
319
320    fn process_reads(&mut self, qconn: &mut QuicheConnection) -> QuicResult<()> {
321        log::trace!("h3i: process_reads");
322
323        // This is executed in process_reads so that work_loop can clear any waits
324        // on the current event loop iteration - if it was in process_writes, we
325        // could potentially miss waits and hang the client.
326        self.register_waits();
327
328        let stream_events = parse_streams(qconn, self);
329        for event in stream_events {
330            self.waiting_for_responses.remove_wait(event);
331        }
332
333        Ok(())
334    }
335
336    fn process_writes(&mut self, qconn: &mut QuicheConnection) -> QuicResult<()> {
337        log::trace!("h3i: process_writes");
338
339        if !self.waiting_for_responses.is_empty() {
340            log::debug!(
341                "awaiting responses on streams {:?}, skipping further action",
342                self.waiting_for_responses
343            );
344
345            return Ok(());
346        }
347
348        // Re-create the iterator so we can mutably borrow the stream parser map
349        let iter = self.actions.clone().into_iter().skip(self.actions_executed);
350
351        for action in iter {
352            match action {
353                Action::SendFrame { .. } |
354                Action::StreamBytes { .. } |
355                Action::ResetStream { .. } |
356                Action::StopSending { .. } |
357                Action::OpenUniStream { .. } |
358                Action::ConnectionClose { .. } |
359                Action::SendHeadersFrame { .. } => {
360                    if self.should_fire() {
361                        // Reset the fire time such that the next action will
362                        // still fire.
363                        self.next_fire_time = Instant::now();
364
365                        execute_action(&action, qconn, self.stream_parsers_mut());
366                        self.actions_executed += 1;
367                    } else {
368                        break;
369                    }
370                },
371                Action::Wait { .. } => {
372                    // Break out of the write phase if we see a wait, since waits
373                    // have to be registered in the read
374                    // phase. The actions_executed pointer will be
375                    // incremented there as well
376                    break;
377                },
378                Action::FlushPackets => {
379                    self.actions_executed += 1;
380                    break;
381                },
382            }
383        }
384
385        Ok(())
386    }
387
388    async fn wait_for_data(
389        &mut self, qconn: &mut QuicheConnection,
390    ) -> QuicResult<()> {
391        log::trace!("h3i: wait_for_data");
392
393        let sleep_fut = if !self.should_fire() {
394            sleep_until(self.next_fire_time)
395        } else {
396            // If we have nothing to send, allow the IOW to resolve wait_for_data
397            // on its own (whether via Quiche timer or incoming data).
398            sleep(Duration::MAX)
399        };
400
401        select! {
402            rx = &mut self.close_trigger_seen_rx, if !self.close_trigger_seen_rx.is_terminated() => {
403                // NOTE: wait_for_data can be called again after all close triggers have been seen,
404                // depending on how long it takes quiche to mark the connection as closed.
405                // Therefore we can't re-poll the receiver or we'd panic.
406                if rx.is_ok() {
407                    // TODO: customizable close trigger frames
408                    let _ = qconn.close(true, quiche::h3::WireErrorCode::NoError as u64, b"saw all expected frames");
409                }
410            }
411            _ = sleep_fut => {}
412        }
413
414        Ok(())
415    }
416
417    fn buffer(&mut self) -> &mut [u8] {
418        &mut self.buffer
419    }
420
421    fn on_conn_close<M: Metrics>(
422        &mut self, qconn: &mut QuicheConnection, _metrics: &M,
423        _work_loop_result: &QuicResult<()>,
424    ) {
425        let _ = self
426            .record_tx
427            .send(ConnectionRecord::Close(ConnectionCloseDetails::new(qconn)));
428
429        let _ = self
430            .record_tx
431            .send(ConnectionRecord::ConnectionStats(qconn.stats()));
432
433        let conn_path_stats = qconn.path_stats().collect::<Vec<PathStats>>();
434        let _ = self
435            .record_tx
436            .send(ConnectionRecord::PathStats(conn_path_stats));
437    }
438}
