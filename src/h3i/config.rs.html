27//! Configuration for the h3i client.
28use std::io;
29
30/// Server details and QUIC connection properties.
31#[derive(Clone)]
32pub struct Config {
33    /// A string representing the host and port to connect to using the format
34    /// `<host>:<port>`.
35    pub host_port: String,
36    /// If the SNI should be omitted during the TLS handshake.
37    pub omit_sni: bool,
38    /// Set a specific IP address to connect to, rather than use DNS resolution.
39    pub connect_to: Option<String>,
40    /// The source port to use when connecting to a server.
41    pub source_port: u32,
42    /// Whether to verify the server certificate.
43    pub verify_peer: bool,
44    /// The QUIC idle timeout value in milliseconds.
45    pub idle_timeout: u64,
46    /// Flow control limit for the connection in bytes
47    pub max_data: u64,
48    /// Flow control limit for locally-initiated bidirectional streams in bytes.
49    pub max_stream_data_bidi_local: u64,
50    /// Flow control limit for remotely-initiated bidirectional streams in
51    /// bytes.
52    pub max_stream_data_bidi_remote: u64,
53    /// Flow control limit for unidirectional streams in bytes.
54    pub max_stream_data_uni: u64,
55    /// Maximum count for concurrent remotely-initiated bidirectional streams.
56    pub max_streams_bidi: u64,
57    /// "Maximum count for concurrent remotely-initiated unidirectional
58    /// streams".
59    pub max_streams_uni: u64,
60    /// Receiver window limit for the connection in bytes.
61    pub max_window: u64,
62    /// Receiver window limit for a stream in bytes.
63    pub max_stream_window: u64,
64    /// Set the session to attempt resumption.
65    pub session: Option<Vec<u8>>,
66}
67
68impl Config {
69    /// Construct a new config object with default values.
70    pub fn new() -> Self {
71        Self::default()
72    }
73
74    pub fn with_host_port(mut self, host_port: String) -> Self {
75        self.host_port = host_port;
76        self
77    }
78
79    pub fn omit_sni(mut self) -> Self {
80        self.omit_sni = true;
81        self
82    }
83
84    pub fn with_connect_to(mut self, connect_to: String) -> Self {
85        self.connect_to = Some(connect_to);
86        self
87    }
88
89    pub fn with_source_port(mut self, port: u32) -> Self {
90        self.source_port = port;
91        self
92    }
93
94    pub fn verify_peer(mut self, verify_peer: bool) -> Self {
95        self.verify_peer = verify_peer;
96        self
97    }
98
99    pub fn with_idle_timeout(mut self, idle_timeout: u64) -> Self {
100        self.idle_timeout = idle_timeout;
101        self
102    }
103
104    pub fn with_max_data(mut self, max_data: u64) -> Self {
105        self.max_data = max_data;
106        self
107    }
108
109    pub fn with_max_stream_data_bidi_local(
110        mut self, max_stream_data_bidi_local: u64,
111    ) -> Self {
112        self.max_stream_data_bidi_local = max_stream_data_bidi_local;
113        self
114    }
115
116    pub fn with_max_stream_data_bidi_remote(
117        mut self, max_stream_data_bidi_remote: u64,
118    ) -> Self {
119        self.max_stream_data_bidi_remote = max_stream_data_bidi_remote;
120        self
121    }
122
123    pub fn with_max_stream_data_uni(mut self, max_stream_data_uni: u64) -> Self {
124        self.max_stream_data_uni = max_stream_data_uni;
125        self
126    }
127
128    pub fn with_max_streams_bidi(mut self, max_streams_bidi: u64) -> Self {
129        self.max_streams_bidi = max_streams_bidi;
130        self
131    }
132
133    pub fn with_max_streams_uni(mut self, max_streams_uni: u64) -> Self {
134        self.max_streams_uni = max_streams_uni;
135        self
136    }
137
138    pub fn with_max_window(mut self, max_window: u64) -> Self {
139        self.max_window = max_window;
140        self
141    }
142
143    pub fn with_max_stream_window(mut self, max_stream_window: u64) -> Self {
144        self.max_stream_window = max_stream_window;
145        self
146    }
147
148    pub fn build(self) -> Result<Self, io::Error> {
149        if self.host_port.is_empty() {
150            return Err(io::Error::new(
151                io::ErrorKind::InvalidInput,
152                "Must provide a <host:port> to connect".to_string(),
153            ));
154        }
155
156        Ok(Config {
157            host_port: self.host_port,
158            omit_sni: self.omit_sni,
159            connect_to: self.connect_to,
160            source_port: self.source_port,
161            verify_peer: self.verify_peer,
162            idle_timeout: self.idle_timeout,
163            max_data: self.max_data,
164            max_stream_data_bidi_local: self.max_stream_data_bidi_local,
165            max_stream_data_bidi_remote: self.max_stream_data_bidi_remote,
166            max_stream_data_uni: self.max_stream_data_uni,
167            max_streams_bidi: self.max_streams_bidi,
168            max_streams_uni: self.max_streams_uni,
169            max_window: self.max_window,
170            max_stream_window: self.max_stream_window,
171            session: None,
172        })
173    }
174}
175

177    fn default() -> Self {
178        // Values mirror config_from_clap()
179        Self {
180            host_port: "".to_string(),
181            omit_sni: false,
182            connect_to: None,
183            source_port: 0,
184            verify_peer: true,
185            idle_timeout: 5000,
186            max_data: 10000000,
187            max_stream_data_bidi_local: 10000000,
188            max_stream_data_bidi_remote: 10000000,
189            max_stream_data_uni: 10000000,
190            max_streams_bidi: 100,
191            max_streams_uni: 100,
192            max_window: 25165824,
193            max_stream_window: 16777216,
194            session: None,
195        }
196    }
