46#[cfg(target_os = "linux")]
47pub fn is_nonblocking(fd: &impl AsRawFd) -> std::io::Result<bool> {
48    let flags = unsafe { libc::fcntl(fd.as_raw_fd(), libc::F_GETFL) };
49
50    if flags == -1 {
51        return Err(std::io::Error::last_os_error());
52    }
53
54    if flags & libc::O_NONBLOCK != 0 {
55        Ok(true)
56    } else {
57        Ok(false)
58    }
59}
60
61#[cfg(all(unix, not(target_os = "linux")))]
62pub fn is_nonblocking(_fd: &impl AsRawFd) -> std::io::Result<bool> {
63    Ok(true)
